\section{Skew Equations}
We will justify and show the four equations used in Lemma \ref{skew rel} to narrow our search for these skew axial algebras. Although they do not provide much use to understanding how these algebras could be constructed, they do make the proof easier.

First, from Lemma \ref{sigma2}, we must have that $Q=R$ and solving for $\lmf_2$, we get
\begin{eqnarray}\label{l2f}
\al\bt(\al-\bt)\lmf_2&=&2(3\al^2-4\al\bt-\al+2\bt)(\lmf_1)^2+2\al(\al-1)\lm_1\lmf_1\nonumber\\
&+&2\al(-2\al-2\bt+1)(\al-\bt)\lmf_1-4\bt(\al-1)(\al-\bt)\lm_1\nonumber\\
&+&\bt(\al-\bt)(4\al^2-3\al+2\bt).
\end{eqnarray}
Suppose $v$ is an $\mu$-eigenvector of an axis, $x$, where $\mu\neq1$. Then the projection on that axis should be equal to 0; that is, $\lm_x(v)=0$. Coincidentally, nearly all of the eigenvectors of $a$ and $b$ stated satisfy that rule. However we have
\begin{equation}\label{proof1}
 0=\lm_b\left(-\frac{P}{\bt}a+Pb+c\right) = -\frac{P}{\bt}\lmf_1+P+\lmf_2 \iff \lmf_2=-\frac{P}{\bt}\gm^f.
\end{equation}
Using notation from Lemma 4.4 in \cite{franchi20211}, let $w=w_1=\frac{1}{2}(b-c)$ to be the $\bt$-component of $b$ in $A_\bt(a)$. Notice that $a$ and $\sg$ has no $\bt$ component and $c$ has $-w$ in its decomposition. Let $z:= ab -\al a - \bt b=\sg -(\al-\bt)a\in A_{\{0,1\}}(b)$.
Notice
\[ az = a(\sg -(\al-\bt)a)=(\dt -(\al-\bt))a+\frac{1}{2}\bt(\al-\bt)(b+c)+(\al-\bt)\sg\]
and so
\begin{eqnarray*}
b(az) &=& (\dt -(\al-\bt))ab+\frac{1}{2}\bt(\al-\bt)(b+bc)+(\al-\bt)b\sg\\
& =& [ \bt \dt -\bt(\al-\bt)+\frac{1}{2}\bt(\al-\bt)P+\bt(\al-\bt)^2]a\\
&+&[ \bt \dt -\bt(\al-\bt)+\frac{1}{2}\bt(\al-\bt)+(\al-\bt)\dt^f]b\\
&+&[\dt -(\al-\bt)+\frac{1}{2}(\al-\bt)P+(\al-\bt)^2]\sg.
\end{eqnarray*}
Hence
\[ [(b(az)]_\bt = \left(\bt \dt -\bt(\al-\bt)+\frac{1}{2}\bt(\al-\bt)+(\al-\bt)\dt^f\right)w.\]
We also have 
\begin{eqnarray*}
(ba)z&=&(\sg+\bt a +\bt b)(\sg -(\al-\bt)a)\\
&=& \sg^2 -(\al-2\bt)a\sg +\bt b\sg -\bt(\al-\bt)a - \bt(\al-\bt)ab\\
&=& [...]a +[\te -\frac{1}{2}\bt(\al-\bt)(\al-2\bt) +\bt\dt^f-\bt^2(\al-\bt)]b\\
&+&[\te -\frac{1}{2}\bt(\al-\bt)(\al-2\bt)]c+[...]\sg\\
&=&[...]a +[\te -\frac{1}{2}\bt(\al-\bt)(\al-2\bt) +\bt\dt^f-\bt^2(\al-\bt)]b\\
&+&[\te -\frac{1}{2}\bt(\al-\bt)(\al-2\bt)]c+[...]\sg.
\end{eqnarray*}
Therefore
\begin{eqnarray*}
[(ba)z]_\bt &=& [\te -\frac{1}{2}\bt(\al-\bt)(\al-2\bt) +\bt\dt^f-\bt^2(\al-\bt)]w\\
& -&[\te -\frac{1}{2}\bt(\al-\bt)(\al-2\bt)]w\\
&=&(\bt\dt^f-\bt^2(\al-\bt))w.
\end{eqnarray*}
By Seress Lemma, we have $0=(ba)z-b(az)$ and so $0=[(ba)z]_\bt-[b(az)]_\bt$. Looking at the coefficient of $w$, we have
\begin{eqnarray*} 
0&=& (\bt\dt^f-\bt^2(\al-\bt))\\
& -& \left(\bt \dt -\bt(\al-\bt)+\frac{1}{2}\bt(\al-\bt)+(\al-\bt)\dt^f\right)\\
&=&-\bt^2(\al-\bt) -\bt\dt+\frac{1}{2}\bt(\al-\bt)-(\al-2\bt)\dt^f.
\end{eqnarray*}
Rearranging we get
\begin{eqnarray}\label{proof2}
\bt\dt =\frac{1}{2}\bt(\al-\bt)-\bt^2(\al-\bt)-(\al-2\bt)\dt^f.
\end{eqnarray}
Let $x:=Pa+\frac{P}{\bt}\sg -\al c=c(b-\al)\in A_{\{0,1\}}(b)$. We have
\begin{eqnarray*}
ax &=& Pa +\frac{P}{\bt}\left(\dt a + \frac{1}{2}\bt(\al-\bt)(b+c) +(\al-\bt)\sg\right)\\
& -&\al(\bt a +\bt c +\sg)\\
&=&\left[P +\frac{P}{\bt}\dt -\al\bt\right]a+\left[\frac{1}{2}(\al-\bt)P\right]b\\
&+&\left[\frac{1}{2}(\al-\bt)P-\al\bt\right]c+\left[\frac{P}{\bt}(\al-\bt)-\al\right]\sg.\\
\end{eqnarray*}
Therefore
\begin{eqnarray*}
b(ax) &=&\left[P +\frac{P}{\bt}\dt -\al\bt\right](\bt a +\bt b+\sg)+\left[\frac{1}{2}(\al-\bt)P\right]b\\ 
&+&\left[\frac{1}{2}(\al-\bt)P-\al\bt\right]\left(P a+\frac{P}{\bt}\sg\right) \\
&+&\left[\frac{P}{\bt}(\al-\bt)-\al\right]\left(\bt(\al-\bt)a+\dt^f b +(\al-\bt)\sg\right)\\
&=&\left[\bt P +\dt P -\al\bt^2+\frac{1}{2}(\al-\bt)P+\frac{P}{\bt}(\al-\bt)\dt^f -\al\dt^f\right]b\\
& +&[...]a +[...]\sg.
\end{eqnarray*}
Hence 
\[[b(ax)]_\bt = \left(\bt P +\dt P -\al\bt^2+\frac{1}{2}(\al-\bt)P+\frac{P}{\bt}(\al-\bt)\dt^f -\al\dt^f\right)w.\]
We also have
\begin{eqnarray*}
(ba)x&=&\left(\bt a +\bt b +\sg\right)\left(Pa+\frac{P}{\bt}\sg -\al c\right)\\
&=&2Pa\sg +\frac{P}{\bt}\sg^2 -\al c \sg + \bt P a -\al\bt ac\\
&+&\bt P ab +Pb\sg -\al\bt bc\\
&=&[...]a + \left[\bt(\al-\bt)P+\frac{P}{\bt}\te+\bt^2P +P\dt^f\right]b\\
&+&\left[\bt(\al-\bt)P+\frac{P}{\bt}\te-\al\dt^f -\al\bt^2\right]c +[...]\sg
\end{eqnarray*}
and so 
\begin{eqnarray*}
    [(ba)x]_\bt &=& \left(\bt(\al-\bt)P+\frac{P}{\bt}\te+\bt^2P +P\dt^f\right)w\\
& -&\left(\bt(\al-\bt)P+\frac{P}{\bt}\te-\al\dt^f -\al\bt^2\right)w\\
    &=& \left(\bt^2P +P\dt^f+\al\dt^f +\al\bt^2 \right)w.
\end{eqnarray*} 
By Seress Lemma, $0=[b(ax)]_\bt-[(ba)x]_\bt$ and looking at the coefficient of $w$, we get 
\begin{eqnarray*}
0&=&[b(ax)]_\bt-[(ba)x]_\bt\\
&=&\left(\bt P +\dt P -\al\bt^2+\frac{1}{2}(\al-\bt)P+\frac{P}{\bt}(\al-\bt)\dt^f -\al\dt^f\right)\\
&-&\left(\bt^2P +P\dt^f+\al\dt^f +\al\bt^2 \right)\\ 
&=&\left(\frac{P}{\bt}\left[\bt^2 +\bt\dt+\frac{1}{2}\bt(\al-\bt)+(\al-2\bt)\dt^f-\bt^3\right]-2\al(\dt^f+\bt^2)\right).
\end{eqnarray*}
From equation (\ref{proof2}), we get that
\begin{eqnarray*}
0&=&\frac{P}{\bt}\left[\bt^2 -\bt^2(\al-\bt) +\frac{1}{2}\bt(\al-\bt)-(\al-2\bt)\dt^f\right.\\
&+&\left.\frac{1}{2}\bt(\al-\bt)+(\al-2\bt)\dt^f-\bt^3\right]-2\al(\dt^f+\bt^2)\\
&=&\frac{P}{\bt}\left[\bt^2 -\bt^2(\al-\bt) +\bt(\al-\bt)-\bt^3\right]-2\al(\dt^f+\bt^2)\\
&=&\frac{P}{\bt}\al\bt\left[1-\bt\right]-2\al(\dt^f+\bt^2).
\end{eqnarray*}
Hence we have 
\begin{equation}\label{proof3}
\frac{1}{2}(1-\bt)P=(\al-1)\gm^f.
\end{equation}
