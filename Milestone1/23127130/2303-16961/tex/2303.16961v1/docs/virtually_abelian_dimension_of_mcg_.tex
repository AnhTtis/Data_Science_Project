\section{Vitually abelian dimension of MCG}



\begin{theorem}
Let $S$ be a connected, closed, oriented  and possibly with a finite number of punctures. Denoted by $G=\mcg(S)$. Then $\gd_{\calF_2}(G)\leq \gd_{\calF_1}(G)+1$.
\end{theorem}
\begin{proof}
 Let $\sim$ be  the equivalence relation in $\calF_2-\calF_1$ defined by commensurability. Let $I$ be  a set complete of representatives of the conjugacy classes in  $\calF_2-\calF_1/\sim$. Then by \cite[Theorem 2.3]{LW12} the following homotopy  $G$-push-out give a model of $E_{\calF_2}G$
 
   \[
\begin{tikzpicture}
  \matrix (m) [matrix of math nodes,row sep=3em,column sep=4em,minimum width=2em]
  {
     \displaystyle\bigsqcup_{[H]\in I} G\times_{N_{G}[H]}E_{\calF_1}N_{G}[H] & E_{\calF_1}G \\
      \displaystyle\bigsqcup_{[H]\in I} G\times_{N_{G}[H]}E_{ \calF_2[H]}N_{G}[H] & X \\};
  \path[-stealth]
    (m-1-1) edge node [left] {$\displaystyle\bigsqcup_{[H]\in I}id_{G}\times_{N_G}f_{[H]}$} (m-2-1) (m-1-1.east|-m-1-2) edge  node [above] {$i$} (m-1-2)
    (m-2-1.east|-m-2-2) edge node [below] {} (m-2-2)
    (m-1-2) edge node [right] {} (m-2-2);
\end{tikzpicture}
\]
the dimension of $X$ is bound by the $\max \{\gd_{\calF_1}(N_{G}[H])+1, \gd_{\calF_1}(G), \gd_{\calF_2[H]}(N_{G}[H])\}_{H\in I}$. Then 
\begin{equation} \label{eq1}
\begin{split}
\gd_{\calF_2}(G) & \leq \max \{\gd_{\calF_1}(N_{G}[H])+1, \gd_{\calF_1}(G), \gd_{\calF_2[H]}(N_{G}[H])\}_{H\in I} \\
 & \leq \max \{ \gd_{\calF_1}(G)+1, \gd_{\calF_2[H]}(N_{G}[H])\}_{H\in I}
\end{split}
\end{equation}
the second inequality it follows from $\gd_{\calF_1}(N_{G}[H])\leq \gd_{\calF_1}(G)$. Now we give a bound of $\gd_{\calF_2[H]}(N_{G}[H])$. By \cref{norrmalizer:commensurator:equal} there is a subgroup $L\leq H$ of finite index  such that $N_{G}[L]=N_{G}(L)$, then we can replace $N_{G}[H]$ by $N_{G}[L]=N_{G}(L)$. The family $$\calF_2[L]=\{K\subseteq N_G[L]| K\in  \calF_2-\calF_1, [K]=[L]\}\cup (\calF_1\cap N_G[L])$$ can be write as the union of two families $\calF_2[L]=\calG \cup \calF$ where $\calG$ is the family generated by  $\{K\subseteq N_G[L]| K\in  \calF_2-\calF_1, [K]=[L]\}$ union the family $\calF_0$ of $N_G[L]$, and $\calF$ is family $(\calF_1\cap N_G[L])$. By \cite[Lemma 4.4]{MR2832565} the following homotopy  $N_G(L)$-push-out give a model for $E_{\calF_2[L]}N_G(L)$

  \[
\begin{tikzpicture}
  \matrix (m) [matrix of math nodes,row sep=3em,column sep=4em,minimum width=2em]
  {
     E_{\calF \cap \calG}N_{G}(L) & E_{\calF}N_{G}(L) \\
      E_{\calG}N_{G}(L) & Y \\};
  \path[-stealth]
    (m-1-1) edge node [left] {} (m-2-1) (m-1-1.east|-m-1-2) edge  node [above] {} (m-1-2)
    (m-2-1.east|-m-2-2) edge node [below] {} (m-2-2)
    (m-1-2) edge node [right] {} (m-2-2);
\end{tikzpicture}
\]

Note that   $\dim(Y) \leq\max \{\gd_{\calF}(N_{G}(L)), \gd_{\calF \cap \calG}(N_{G}(L)), \gd_{\calG}(N_{G}(L))\}$. Then 
\begin{equation} \label{eq2}
\begin{split}
\gd_{\calF_2[L]}(N_{G}[L]) & \leq\max \{\gd_{\calF}(N_{G}(L)), \gd_{\calF \cap \calG}(N_{G}(L)), \gd_{\calG}(N_{G}(L))\} \\
 & \leq \max \{ \gd_{\calF_1}(G), \gd_{\calF \cap \calG}(N_{G}(L)), \gd_{\calG}(N_{G}(L))\}.
\end{split}
\end{equation}
Now we prove the followings inequalities
\begin{enumerate}[i)]
    \item $\gd_{\calG}(N_{G}(L))\leq \vcd W(L)$.
    \item $\gd_{\calF \cap \calG}(N_{G}(L)) \leq \vcd W(L)+3$
\end{enumerate}
 First we going to show $\gd_{\calG}(N_{G}(L))\leq \vcd W(L)$. A model for $E_{\calF_0}W(L)$ is a model for $E_{\calG}N_{G}(L)$ via the action given by the projection $N_{G}(L)\to N_{G}(L)/L$. 

By \cref{Normalizer:commensurator:free:abelian:subgroup} we have the following   short exact sequence 

\begin{equation}\label{s.e.s:normalaizer:1}
1\to \Z^n \cap N_{\mcg(S)}(L) \to  N_{\mcg(S)}(L)\xrightarrow[]{\rho_{\sigma}} Q\to 1\end{equation}
with $Q\subseteq \mcg(\displaystyle\bigsqcup_{i=1}^{a}S_i)\times A$ where $A$ is virtually abelian group of rank almost $l-a$. From \cref{s.e.s:normalaizer:1} we have the next short exact sequence 

\begin{equation}\label{s.e.s:weil:1}
1\to  (\Z^n \cap N_{G}(L))/(L\cap(\Z^n \cap N_{G}(L))) \to  W(L)\xrightarrow[]{\rho_{\sigma}} Q/\rho_{\sigma}(L)\to 1\end{equation}

We proof that $W(L)$ has a model for  $E_{\calF_0}W(L)$  of dimension $\vcd(W(L))$, by \cref{pro:key:3} it is enough to show that   $Q/\rho_{\sigma}(L)$ has a model for $E_{\calF_0}Q/\rho_{\sigma}(L)$ of dimension $\vcd Q/\rho_{\sigma}(L)$. Now note that   $Q/\rho_{\sigma}(L)= \mcg(\bigsqcup_{i=1}^{a}S_i)\times A/\rho_{\sigma}(L)$ then $\gd_{\calF_0}(Q/\rho_{\sigma}(L))= \gd_{\calF_0}(\mcg(\bigsqcup_{i=1}^{a}S_i))+ \gd_{\calF_0}A/\rho_{\sigma}(L)$, by \cite[Proposition 5.3]{Nucinkis:Petrosyan} we have that $\gd_{\calF_0}(\mcg(\bigsqcup_{i=1}^{a}S_i))=\vcd \mcg(\bigsqcup_{i=1}^{a}S_i)$ then by \cref{vcd:s.e.s:1} we have $\gd_{\calF_0}(Q/\rho_{\sigma}(L))= \vcd  Q/\rho_{\sigma}(L)$ as desired.


Now we bound $\gd_{\calF \cap \calG}(N_{G}[H])$. Note that $\calF \cap \calG \subset \calG$, then  by   \cref{pro:key:2} we need to find a integer $d>0$ such that for any $S\in \calG$ we have $\gd_{\calF\cap \calG\cap 
S}S\leq d$. Let $S\in \calG$ then we have three cases (i) $S$ is finite , (ii) $S$ is virtually $\Z$ or (iii) $S$ is virtually $\Z^2$. 

In the case (i) and (ii) we have $\gd_{\calF\cap \calG\cap 
S}S= 0$ since $S$ is virtually cyclic  then $S\in \calF$ but also $S\in \calG$, it follows that $S\in \calF\cap \calG\cap 
S$. In the case (iii) we have $\gd_{\calF\cap \calG\cap 
S}S= 3$ this because we going to proof that  $\calF\cap \calG\cap 
S$ is the family of virtually cyclic subgroups $\calH$ of $S$. The contention $\calF\cap \calG\cap S \subset \calH$ is clear. Remains to show that $\calH \subset \calF\cap \calG\cap S$. The fact that $S\in \calG$ and $S$ is virtually $\Z^{2}$ implies that $S$ is commensurable with $H$, then we have $N_{G}[S]=N_{G}[H]$. Let $L\in \calH$ by the before we have $L\in \calF$,  by hypothesis $S\in \calG$ and $\calG$ is a family implies also that $L\in \calG$. It follows that $L\in \calF\cap \calG\cap S$.  
Therefore we have by \cref{pro:key:2} that $\gd_{\calF \cap \calG}(N_{G}[H])\leq \vcd W(L)+3$.

Thus, by \cref{eq2} we have 
\begin{equation} \label{eq3}
\begin{split}
\gd_{\calF_2[H]}(N_{G}[H]) &=\gd_{\calF_2[L]}(N_{G}[L]) \\ 
 & \leq \max \{ \gd_{\calF_1}(G), \gd_{\calF \cap \calG}(N_{G}(L)), \gd_{\calG}(N_{G}(L))\}\\
 & \leq \max \{ \gd_{\calF_1}(G), \vcd W(L)+3, \vcd W(L)\}\\
 &= \max \{ \gd_{\calF_1}(G), \vcd W(L)+3\}\\
 &= \max \{ \gd_{\calF_1}(G), \vcd N_{G}(L)+1\}\\
 &\leq \max \{ \gd_{\calF_1}(G), \vcd G+1\}\\
&=  \gd_{\calF_1}(G).
\end{split}
\end{equation}

Finally by \cref{eq1} we have $\gd_{\calF_2}(G)\leq \gd_{\calF_1}(G)+1$.


\end{proof}
