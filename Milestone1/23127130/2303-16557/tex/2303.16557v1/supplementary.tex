  \centering
     \caption{Evaluation metric for comparing the performance of SAT and others. $y_i$,$\hat{y}_i$ deontes the ground truth and predicted value respectively. $\theta$ is the hyperparameter where it decides that the difference value between prediction and ground truth label belongs to a certain range.}\label{tab:data-statistic}
    \begin{tabular}{c|c}
    \toprule
        metric & equation \\
        \midrule
        \hline
        MAE (Mean Absolute Error)
       & $\frac{1}{N} \sum_{i=1}^{N}|y_i - \hat{y}_i|$ \\
       \hline
        CS{($\theta$)}
       & $\frac{1}{N} \sum_{i=1}^{N} \mathbbm{1}(|y_i - \hat{y}_i| \leq \theta) \times 100\%$ \\
        \bottomrule
      \end{tabular}
