\begin{tikzpicture}[
    node distance=1cm,
    vnode/.style = {vertex, minimum size=1.2em}
]
    \node[vnode] (v1) {};
    \node[vnode, below left of=v1] (v2) {};
    \node[vnode, below right of=v1] (v3) {};

    \path[draw, edge]
        (v1) edge[plus] (v2)
        (v2) edge[plus] (v3)
        (v3) edge[plus] (v1);

    \node[label, above of=v1, node distance=2em] (label1) {Balanced};

    \node[vnode, right of=v1, node distance=3cm] (v4) {};
    \node[vnode, below left of=v4] (v5) {};
    \node[vnode, below right of=v4] (v6) {};

    \path[draw, edge]
        (v4) edge[minus] (v5)
        (v4) edge[plus] (v6)
        (v5) edge[plus] (v6);

    \node[label, above of=v4, node distance=2em] (label2) {Unbalanced};
    \node[label, below of=v3, node distance=.75cm, xshift=1cm] (np1) 
        {\# balanced cycles $\lessapprox$ \# unbalanced cycles}
    \node[label, below of=np1, node distance=.5cm, xshift=-.45em, rotate=-90] (to1) {$\to$};
    \node[label, below of=to1, node distance=.5cm] (polarization) {Low polarization};
        
\end{tikzpicture}
