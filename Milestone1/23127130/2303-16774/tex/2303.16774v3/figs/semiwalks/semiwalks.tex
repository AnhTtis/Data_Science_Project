\begin{figure}[t!]
\begin{tikzpicture}[node distance=1.5cm]

    \node[vertex] (i) {$i$};
    \node[vertex, above right of=i] (k) {$k$};
    \node[vertex, above left of=k] (j) {$j$};

    \draw[->, ultra thick]
        (i) edge[plus, bend right] (k)
        (k) edge[minus, bend right] (i);
    \draw[->, ultra thick]
        (k) edge[plus] (j);

    \node[label, right of=k] (map) {\LARGE$\mapsto$};

    \node[vertex, right of=map, yshift=.35cm] (pi1) {$i$};
    \node[vertex, right of=pi1] (pk1) {$k$};
    \node[vertex, right of=pk1] (pj1) {$j$};
    \draw[ultra thick]
        (pi1) edge[plus] (pk1)
        (pk1) edge[plus] (pj1);

    \node[vertex, below of=pi1, node distance=.75cm] (pj2) {$j$};
    \node[vertex, right of=pj2] (pk2) {$k$};
    \node[vertex, right of=pk2] (pi2) {$i$};
    \draw[ultra thick]
        (pj2) edge[plus] (pk2)
        (pk2) edge[plus] (pi2);
    \node[label, above of=pk1, node distance=.75cm] (lab1)
        {Positive semiwalks};

    \node[vertex, right of=pj1] (ni1) {$i$};
    \node[vertex, right of=ni1] (nk1) {$k$};
    \node[vertex, right of=nk1] (nj1) {$j$};
    \draw[ultra thick]
        (ni1) edge[minus] (nk1)
        (nk1) edge[plus] (nj1);

    \node[vertex, below of=ni1, node distance=.75cm] (nj2) {$j$};
    \node[vertex, right of=nj2] (nk2) {$k$};
    \node[vertex, right of=nk2] (ni2) {$i$};
    \draw[ultra thick]
        (nj2) edge[plus] (nk2)
        (nk2) edge[minus] (ni2);
    \node[label, above of=nk1, node distance=.75cm] (lab2)
        {Negative semiwalks};
    
\end{tikzpicture}
\end{figure}
