    \node[vertex] (i) {$i$};
    \node[vertex, above right of=i] (k) {$k$};
    \node[vertex, below right of=k] (j) {$j$};
    \path[draw, dedge]
        (i) edge[plus] (j)
        (j) edge[minus] (k)
        (k) edge[plus, bend right] (i)
        (i) edge[minus, bend right] (k);
        
    \node[draw, ellipse, fit=(i)(j)(k), yshift=-.5em, color=black!20, inner sep=0, outer sep=.5em] 
        (e1) {};

    \node[label, left of=i, node distance=4cm, yshift=1.5cm, outer sep=.5em] (blab)
        {Balanced};
    \node[vertex, below of=blab, node distance=1cm] (k1) {$k$};
    \node[vertex, below left of=k1] (i1) {$i$};
    \node[vertex, below right of=k1] (j1) {$j$};
    \path[draw, edge, latex'-latex']
        (i1) edge[plus] (j1)
        (j1) edge[minus] (k1)
        (k1) edge[minus] (i1);
    \coordinate[right of=k1, node distance=5em] (k1c1);
    \coordinate[right of=k1c1, node distance=3em] (k1c2);
    \path[draw, dedge]
        (k1c2)--(k1c1);

    \node[label, right of=j, node distance=4cm, yshift=1.5cm, outer sep=.5em] (ulab)
        {Unbalanced};
    \node[vertex, below of=ulab, node distance=1cm, xshift=0cm] (k2) {$k$};
    \node[vertex, below left of=k2] (i2) {$i$};
    \path[draw, edge, latex'-latex']
        (i2) edge[minus, bend right] (k2)
        (k2) edge[plus, bend right] (i2);
    \node[vertex, below of=ulab, node distance=1cm, xshift=1.5cm] (k3) {$k$};
    \node[vertex, below left of=k3] (i3) {$i$};
    \node[vertex, below right of=k3] (j3) {$j$};
    \path[draw, edge, latex'-latex']
        (i3) edge[plus] (j3)
        (j3) edge[minus] (k3)
        (k3) edge[plus] (i3);
    \coordinate[left of=k2, node distance=4.5em] (k2c1);
    \coordinate[left of=k2c1, node distance=3em] (k2c2);
    \path[draw, dedge]
        (k2c2)--(k2c1);
    
\end{tikzpicture}
