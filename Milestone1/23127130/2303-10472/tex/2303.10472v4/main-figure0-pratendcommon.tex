

\begin{restatable}[]{lemma}{prAtEndRestatev}\label{thm:prAtEndv}{(\citealt [Lemma 9]{domke_provable_2019})}\label {thm:u_identities} Let \(\rvvu = \left (\rvu _1, \rvu _2, \ldots , \rvu _d\right )\) be a \(d\)-dimensional vector-valued random variable with zero-mean independently and identically distributed components. Then, \begin {alignat*}{2} &\mathbb {E}\rvvu \rvvu ^{\top } &&= \left ( \mathbb {E} \rvu _i^2 \right ) \mI \\ &\mathbb {E}\norm {\rvvu }_2^2 &&= d \, \mathbb {E} \rvu _i^2 \\ &\mathbb {E} \rvvu \left ( 1 + \norm {\rvvu }_2^2 \right ) &&= \left ( \mathbb {E} \rvu _i^3 \right ) \mathbf {1} \\ &\mathbb {E} \rvvu \rvvu ^{\top } \rvvu \rvvu ^{\top } &&= \left ( \left (d - 1\right ) \, {\left ( \mathbb {E} \rvu _i^2 \right )}^2 + \mathbb {E}\rvu _i^4 \right ) \mI . \end {alignat*}\end{restatable}

\begin{restatable}[]{lemma}{prAtEndRestatevi}\label{thm:prAtEndvi}{(\citealt [Lemma 1]{domke_provable_2019})}\label {thm:variational_gradient_norm_identity} Let \(\vt _{\vlambda }: \mathbb {R}^d \mapsto \mathbb {R}^d\) be defined as in \cref {def:reparam} with parameters \(\vlambda = \left (\vm , \mC \right )\). Also, let \(f : \mathbb {R}^d \mapsto \mathbb {R} \) be some differentiable function. Then, \begin {alignat*}{2} \norm {\nabla _{\vlambda } f\left ( \vt _{\vlambda }\left (\vu \right ) \right ) }_2^2 = \mathbb {E}\norm {\nabla f\left ( \vt _{\vlambda }\left (\vu \right ) \right ) }_2^2 \left (1 + \norm {\vu }_2^2\right ). \end {alignat*}\end{restatable}

\begin{restatable}[]{lemma}{prAtEndRestatevii}\label{thm:prAtEndvii}{(\citealt [Lemma 1]{domke_provable_2019})}\label {thm:reparam_u_identity} Let \(\vt _{\vlambda }: \mathbb {R}^d \mapsto \mathbb {R}^d\) be defined as in \cref {def:reparam} with parameters \(\vz = \left (\vm , \mC \right )\). Also, let \(\vz \in \mathbb {R}^d\) be some vector and \(\rvvu = \left (\rvu _1, \ldots , \rvu _d \right )\) be a vector-valued random variable sampled as \(\rvvu \sim \varphi \), where \(\varphi \) is defined as in \cref {assumption:symmetric_standard} with kurtosis \(\kappa = \mathbb {E}\rvu _i^4\). Then, \begin {alignat*}{2} &\mathbb {E} \norm {\,\vt _{\vlambda }\left (\rvvu \right ) - \vz \,}_2^2 \left (1 + \norm {\rvvu }_2^2\right ) \\ &\;= \left (d+1\right ) \norm {\vm - \vz }_2^2 + \left (d + \kappa \right ) \norm {\mC }^2_{\mathrm {F}}. \end {alignat*}\end{restatable}
