\makeatletter\Hy@SaveLastskip\label{proofsection:prAtEndviii}\ifdefined\pratend@current@sectionlike@label\immediate\write\@auxout{\string\gdef\string\pratend@section@for@proofviii{\pratend@current@sectionlike@label}}\fi\Hy@RestoreLastskip\makeatother\begin{proof}[Proof]\phantomsection\label{proof:prAtEndviii}This proof uses the smoothness of \(f_{\mathrm {KL}}\) instead of \(f_{\mathrm {H}}\). That is, \begin {align} \mathbb {E} \norm { \nabla f_{\mathrm {KL}}\left (\vt _{\vlambda }\left (\rvvu \right )\right ) }_2^2 &= \mathbb {E} {\lVert \nabla f_{\mathrm {KL}}\left (\vt _{\vlambda }\left (\rvvu \right )\right ) - \nabla f_{\mathrm {KL}}\left ( \bar {\vzeta }_{\mathrm {KL}} \right ) \rVert }_2^2 \nonumber \shortintertext {applying \cref {eq:thm_upper_bound_smoothness},} &\leq L^2_{\mathrm {KL}} \, \mathbb {E} {\lVert \vt _{\vlambda }\left (\rvvu \right ) - \bar {\vzeta }_{\mathrm {KL}} \rVert }_2^2, \label {eq:thm3_kl_smoothness} \end {align} where \(\bar {\vzeta }_{\mathrm {KL}}\) is a stationary point of \(f_{\mathrm {KL}}\). \par Substituting \cref {eq:thm3_kl_smoothness} in \cref {eq:thm1_parameter_suboptimality_unified}, \begin {alignat}{2} &\mathbb {E}\norm {\nabla _{\vlambda } f_{\mathrm {KL}}\left (\vt _{\vlambda }\left (\rvvu \right )\right )}_2^2 \nonumber \\ &\;\leq L^2_{\mathrm {KL}} C\left (d, \kappa \right ) \mathbb {E} {\Vert \vt _{\vlambda }\left (\rvvu \right ) - \bar {\vzeta }_{\mathrm {KL}} \rVert }_2^2, \nonumber \shortintertext {and by applying \cref {eq:thm_upper_bound_f_quadratic} for \(f_{\mathrm {KL}}\),} &\;= \frac {2 L^2_{\mathrm {KL}} }{\mu _{\mathrm {KL}}} C\left (d, \kappa \right ) \left ( F\left (\vlambda \right ) - F^* \right ) + \frac {2 L^2_{\mathrm {KL}} }{\mu _{\mathrm {KL}}} C\left (d, \kappa \right ) \left ( F^* - f^*_{\text {KL}} \right ). \nonumber \end {alignat} Plugging this to \cref {thm:gradient_variance_general_upper_bound} proves the result.\end{proof}
