	\begin{algorithm}[H]
	\LinesNumbered
	\SetAlgoLined
	\KwIn{finite field size $q$, received message $y$, generator matrix $G$ of linear code $C$.}
	\KwOut{estimated codeword $\hat{x}\in C$ and estimated lattice point $x\in\Lambda_A(C)$.}
	$y=y'\mod q$;\\
	\For{$u = 0$ \KwTo $q-1$}{
		$d_u=(y' - G\odot u)\mods q$;
	}
	$j=\argmin\{\min_{u}||d_u||^2\}$;\\
	$d_{min}=d_j$;\\
	$\hat{x} = (y' - d_{min})\mod q$;\\
	$x = y - d_{min}$;
	\label{algo1}
	\caption{Construction A lattices decoder}
	\end{algorithm}
