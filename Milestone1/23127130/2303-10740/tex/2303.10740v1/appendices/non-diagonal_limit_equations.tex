
\subsection{Roe scheme}

Limit equations at convective limit with convective diffusion for Roe.
\begin{subequations}
    \begin{align}
        \partial_t\ord{p}{0} + \gamma\ord{p}{0}\partial_x\ord{u}{0} & = \gamma\ord{p}{0}\Big( \frac{\mu_{11}}{\ord{\rho}{0}|v|}\partial_{xx}\ord{p}{2} + \mu_{12}\partial_{xx}\ord{u}{0} \Big) \\
        \ord{\rho}{0}\partial_t\ord{u}{0} + \partial_x\ord{p}{2} + \ord{\rho u}{0}\partial_x\ord{u}{0} & = \ord{\rho}{0}\mu_u|u|\partial_{xx}\ord{u}{0} + \mu_{21}\partial_{xx}\ord{p}{2} + \ord{\rho}{0}|v|\mu_{22}\partial_{xx}\ord{u}{0} \\
        \partial_t\ord{v}{0}                           +      \ord{u}{0}\partial_x\ord{v}{0} & = \mu_u|u|\partial_{xx}\ord{v}{0} \\
        \partial_t\ord{s}{0}                           +      \ord{u}{0}\partial_x\ord{s}{0} & = \mu_u|u|\partial_{xx}\ord{s}{0}
    \end{align}
\end{subequations}

Limit equations at acoustic limit with convective diffusion for Roe.
\begin{subequations}
    \begin{align}
        0 & = \gamma\ord{p}{0}\frac{\mu_{11}}{\ord{\rho}{0}|v|}\partial_{xx}\ord{p}{1} \\
        \ord{\rho}{0}\partial_{\tau}\ord{u}{0} + \partial_x\ord{p}{1} & = \mu_{21}\partial_{xx}\ord{p}{1} + \ord{\rho}{0}|v|\mu_{22}\partial_{xx}\ord{u}{0} \\
        \partial_{\tau}\ord{v}{0} & = 0 \\
        \partial_{\tau}\ord{s}{0} & = 0
    \end{align}
\end{subequations}

Limit equations at convective limit with mixed diffusion for Roe.
\begin{subequations}
    \begin{align}
        \partial_t\ord{p}{0} + \gamma\ord{p}{0}\partial_x\ord{u}{0} & =  \gamma\ord{p}{0}\mu_{12}\partial_{xx}\ord{u}{0} \\
        \ord{\rho}{0}\partial_t\ord{u}{0} + \partial_x\ord{p}{2} + \ord{\rho u}{0}\partial_x\ord{u}{0} & = \ord{\rho}{0}\mu_u|u|\partial_{xx}\ord{u}{0} + \mu_{21}\partial_{xx}\ord{p}{2} + \ord{\rho}{0}|v|\mu_{22}\partial_{xx}\ord{u}{0} \\
        \partial_t\ord{v}{0}                           +      \ord{u}{0}\partial_x\ord{v}{0} & = \mu_u|u|\partial_{xx}\ord{v}{0} \\
        \partial_t\ord{s}{0}                           +      \ord{u}{0}\partial_x\ord{s}{0} & = \mu_u|u|\partial_{xx}\ord{s}{0}
    \end{align}
\end{subequations}

Limit equations at acoustic limit with mixed diffusion for Roe.
\begin{subequations}
    \begin{align}
        \partial_{\tau}\ord{p}{0} + \gamma\ord{p}{0}\partial_x\ord{u}{0} & = \gamma\ord{p}{0}\Big(\frac{\mu_{11}}{\ord{\rho}{0}|v|}\partial_{xx}\ord{p}{1} + \mu_{12}\partial_{xx}\ord{u}{0}\Big) \\
        \ord{\rho}{0}\partial_{\tau}\ord{u}{0} + \partial_x\ord{p}{1} & = \mu_{21}\partial_{xx}\ord{p}{1} \\
        \partial_{\tau}\ord{v}{0} & = 0 \\
        \partial_{\tau}\ord{s}{0} & = 0
    \end{align}
\end{subequations}

\begin{subequations}
    \begin{align}
        \partial_t\ord{p}{0} + \gamma\ord{p}{0}\partial_x\ord{u}{0} & = 0 \\
