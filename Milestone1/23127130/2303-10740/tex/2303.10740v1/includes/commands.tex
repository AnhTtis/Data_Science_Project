
% month in text, year

April\or May \or June\or July\or August\or September\or October\or November\or %

% bolder dot for divergence operator
\newcommand{\dotp}{\ensuremath{\boldsymbol{\cdot}}}

% shorter minus sign for negative indices (normal minus sign is very long)
\def\-{\scalebox{0.4}[1.0]{\(-\)}}

% shortcut for saying on the order of a power of M
\newcommand{\mathcalOM}[1]{\ensuremath{\mathcal{O}(M^{#1})}}

% shortcut for the order coefficient of asymptotic expansion
%\newcommand{\ord}[2]{\ensuremath{#1^{\scriptscriptstyle{(#2)}}}}
\newcommand{\ord}[2]{\ensuremath{#1^{(#2)}}}

% shortcut for the sum over all faces of finite volume cell
\newcommand{\facesum}{\sum^l_{\mathcal{V}(i)}}

% double underline
\newcommand{\uuline}[1]{\underline{\underline{#1}}}

%trig functions for von Neumann symbols
\newcommand{\sk}{\ensuremath{S_k}}
\newcommand{\ck}{\ensuremath{C_k}}
\newcommand{\skt}{\ensuremath{S_{k/2}}}
\newcommand{\ckt}{\ensuremath{C_{k/2}}}

% Word / Character count commands ---------------------------

\usepackage{verbatim}
\newcommand{\quickwordcount}[1]{%
  \immediate\write18{texcount -1 -sum -merge -q #1.tex output.bbl > #1-words.sum }%
  \input{#1-words.sum} words%
}
\newcommand{\quickcharcount}[1]{%
  \immediate\write18{texcount -1 -sum -merge -char -q #1.tex output.bbl > #1-chars.sum }%
  \input{#1-chars.sum} characters (not including spaces)%
}
