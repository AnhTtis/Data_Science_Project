\newcount\@hour
\newcount\@hourmin
\newcount\@minutes

\def\today{\ifcase\month\or
 January\or February\or March\or April\or May\or June\or
 July\or August\or September\or October\or November\or December\fi
 \space\number\day, \number\year}

\newcount\@ye@r


\def\season{\@ye@r=\year\ifnum\@ye@r<2000\advance\@ye@r by -1900\else
\advance\@ye@r by -2000\fi
\ifcase\month\or Spring\or Spring\or Spring\or Spring\or Spring\or Summer\or
 Summer\or Summer\or Fall\or Fall\or Fall\or Fall\fi
 \space'{\ifnum\@ye@r<10\relax0\number\@ye@r\else\relax\number\@ye@r\fi}}
 
\def\calendar{\@ye@r=\year
\ifnum\@ye@r<2000\advance\@ye@r by -1900\else
\advance\@ye@r by -2000\fi\relax\number\month-\ignorespaces
{\ifnum\day<10\relax0\number\day\else\relax\number\day\fi}-\ignorespaces
{\ifnum\@ye@r<10\relax0\number\@ye@r\else\relax\number\@ye@r\fi}\relax}


\def\clock{
\@hour=\time
\divide\@hour by 60
\@hourmin=\@hour
\multiply\@hourmin by 60
\@minutes=\time
\advance\@minutes by -\@hourmin
\ifnum\@hour=0
      \ifnum\@minutes<10
                 12:0\the\@minutes~AM
                       \else 12:\the\@minutes~AM  \fi 
\else  
   \ifnum\@hour<12
           \ifnum\@minutes<10
                 \the\@hour:0\the\@minutes~AM
                       \else \the\@hour:\the\@minutes~AM  \fi
   \else  \ifnum\@hour=12
              \ifnum\@minutes<10
                    \the\@hour:0\the\@minutes~PM
                       \else \the\@hour:\the\@minutes~PM  \fi
           \else \advance\@hour by -12
		    \ifnum\@minutes<10
                    \the\@hour:0\the\@minutes~PM
                       \else \the\@hour:\the\@minutes~PM  \fi \fi \fi \fi \relax}



\def\uclock{

\@minutes=\time

                       \else \the\@hour:\the\@minutes\fi
