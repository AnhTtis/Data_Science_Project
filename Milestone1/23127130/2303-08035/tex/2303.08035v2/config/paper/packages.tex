% these packages are included only for the paper, as most
% of them are incompatible with beamer in presentation-mode

% we have biber here to have presentation-specific configurations
% url,doi,isbn = false remove the doi,url,isbn in the bibliography
\usepackage[%
    % backend to use
    backend=biber,%
    % whether to include backreferences
    backref=false,%
    % use name/numeric for the references
    citestyle=numeric,%
    % to remove the dash appearing when consequent entries have the same authors
    dashed=false,%
    % to use YYYY-MM-dd as date format
    date=iso,%
    % remove DOI if false
    doi=false,%
    % giveninits puts the pointed first name if true
    % instead of writing it whole, e.g. First Last becomes F. Last
    giveninits=true,%
    % use hyperrefs for citations to the bibliography if true
    hyperref=true,%
    % remove ISBN if false
    isbn=false,%
    % maximum number of names to use in bibliography before going for et al.
    maxbibnames=100,%
    % maximum number of names before et al. in citation if not numeric
    maxcitenames=100,%
    % minimum number of names to use in bibliography before going for et al.
    minbibnames=100,%
    % maximum number of names before et al. in citation if not numeric
    mincitenames=1,%
    % required by date=iso
    seconds=true,%
    % to sort the numbers when using \cite with multiple keys
    sortcites=true,%
    % style of bibliography
    style=ieee,%
    % reference sorting in bibliography
    sorting=none,%
    % remove URLs if false
    url=false,%
    % to use YYYY-MM-dd as date format for URLs
    urldate=iso,%
]{biblatex}

% for custom enumerations
\usepackage{enumitem}
