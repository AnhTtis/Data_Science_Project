\newtcolorbox{usecase}[1][]{
  breakable,
  enhanced,
  arc=0pt,
  outer arc=0pt,
  colframe=titlebg,
  colback=titlebg!05,
  overlay unbroken and first={
    \node[
      draw=titlebg,
      fill=titlebg,
      rotate=0,
      anchor=north west,
      text=white,
      font=\bfseries
    ]
    at (frame.north west)  
    {#1};
  }
}
