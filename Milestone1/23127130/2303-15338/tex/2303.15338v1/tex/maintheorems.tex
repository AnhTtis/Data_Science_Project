\section{The main theorems} \label{section_mainthms}
In this section we state the main theorems in a precise form. We first state the following simple Lemma~\ref{prop_compactness_support_general}, which immediately follows from Assumption~\ref{assumption_support}. Lemma~\ref{prop_compactness_support_general} provides a quantitative and uniform bound for the supremum norm $W$ of a polynomial weight $w$ on the support of a solution to the massless Vlasov equation with compactly supported initial data. In the following theorems, we often merely assume the weight $w$ to be bounded in $x$ as in Definition~\ref{boundedness_in_x}, although we mainly have the case of polynomial weights in mind. We prove Lemma~\ref{prop_compactness_support_general} in Lemma~\ref{psupport_bounded} for the subextremal case and in Lemma~\ref{psupport_boundedness_ERN} for the extremal case.

\begin{lem}[Compactness of momentum support] \label{prop_compactness_support_general}
	Let $f_0 \in L^\infty(\mathcal{P}_0)$ satisfy Assumption~\ref{assumption_support} with constants $\rsuppconst$ and $\psuppconst$ and assume that $f$ is the unique solution to the massless Vlasov equation on the subextremal or extremal Reissner--Nordstr\"om exterior such that $f|_{\mathcal{P}_0} = f_0$. Then the momentum support of $f$ remains compact for all times: There exists a constant $C > 0$ such that
	\begin{equation}
		\supp(f) \subset \Bigg\{ (x,p) \in \mathcal{P} : p^{t^*} \leq C \psuppconst, \; \left| p^r \right| \leq \psuppconst, \; \left| \pslash \right|_{\gslash} \leq \frac{\rsuppconst}{r} \psuppconst \Bigg\} \subset \mathcal{P}.
	\end{equation}
	In particular, if $w: \mathcal{P} \rightarrow [0,\infty)$ is bounded polynomially, i.e.~$\left| w(x,p) \right| \leq C' (p^{t^*})^a \left| p^r \right|^b \left| \pslash \right|^c_{\gslash}$ for some constants $C',a,b,c \geq 0$, then we have the following quantitative and uniform bound
	\begin{equation}
		W = \max_{(x,p) \in \supp(f)} \left| w(x,p) \right| \lesssim_{a} \frac{\psuppconst^{a+b+c} \rsuppconst^c}{r^{c}} .
	\end{equation}
	More generally, if $w$ is bounded in $x$ according to Definition~\ref{boundedness_in_x} then
	\begin{equation}
		W = \max_{(x,p) \in \supp(f)} \left| w(x,p) \right| < \infty.
	\end{equation}
\end{lem}

Next we give a precise version of Theorem~\ref{maintheorem} on the subextremal Reissner--Nordstr\"om exterior. In fact, we will only state the theorem for the Schwarzschild spacetime, since the statement and proof readily generalise to the full subextremal Reissner--Nordstr\"om family. The proof is discussed in Section~\ref{section_integralestimate}.

\begin{thm}[Exponential decay on Schwarzschild] \label{maintheorem_precise}
Assume $f_0 \in L^\infty(\mathcal{P}_0)$ satisfies Assumption~\ref{assumption_support} with constants $\rsuppconst$ and $\psuppconst$ and let $f$ be the unique solution to the massless Vlasov equation on the Schwarzschild exterior such that $f|_{\mathcal{P}_0} = f_0$. Let $w: \mathcal{P} \rightarrow [0,\infty)$ be smooth and bounded in $x$ according to Definition~\ref{boundedness_in_x} so that Lemma~\ref{prop_compactness_support_general} applies and in particular $W < \infty$.

There exist dimensionless constants $C,C_0,\decayrate>0$ such that if $\tauzero = C_0 \left( m + \frac{\rsuppconst^2}{m} \right)$, then for times $\tau(x) \geq \tauzero$ the moment with weight $w$ decays at an exponential rate:
\begin{equation}
		\int_{\mathcal{P}_x} w f \, d \mu_x \leq C W \| f_0 \|_{L^\infty} \psuppconst^2 \rsuppconst^2 \frac{1}{r^2} e^{-\frac{\decayrate}{m} (\tau(x) - \tauzero)}.
\end{equation}

Moreover, we have the following refinements: If $0 < \delta < 1$ and $\tau(x) \geq \tauzero - \ln \delta'$ then in the region where $r \notin [(3-\delta)m,(3+\delta)m]$ the rate of decay may be improved to
\begin{equation}
	\int_{\mathcal{P}_x} w f \, d \mu_x \leq C W \| f_0 \|_{L^\infty} \psuppconst^2 \rsuppconst^2 \frac{1}{r^2} e^{-\frac{2 \decayrate}{m} (\tau(x) - \tauzero)}.
\end{equation}
In addition, for certain weights, an improved decay estimate holds. Let $a,b > 0$ and consider
\begin{equation}
	w_{a,b} = \left| \left| p^r \right| - \sqrt{\frac{1}{27} - \frac{m^2}{r^2} \OsqS} \frac{L}{m} \right|^a \left| p^r \right|^b,
\end{equation}
and define the constant $\decayrate_{a,b} = \decayrate \min \left\{ 1+a, 3 + 2 b \right\}$ with $\decayrate$ as above. Then the following improved decay estimate holds uniformly in $r \geq 2m$,
\begin{equation}
	\int_{\mathcal{P}_x} w_{a,b} f \, d \mu_x \leq C^{2 + a + b} \| f_0 \|_{L^\infty} \frac{\rsuppconst^{2+a} \psuppconst^{2 + a + b}}{m^a} \frac{1}{r^2} e^{- \frac{\decayrate_{a,b}}{m} (\tau(x)-\tauzero)}.
\end{equation}
If $x \in \Mrn$ is chosen along the photon sphere as $\tau \rightarrow \infty$, i.e.~$r = 3m$ remains fixed, then this estimate may be further improved to
\begin{equation}
	\int_{\mathcal{P}_x} \left| p^r \right|^\kappa f \, d \mu_x \Big|_{r=3m} \leq C^{2+\kappa} \| f_0 \|_{L^\infty} \frac{\rsuppconst^{2+\kappa} \psuppconst^{2+\kappa}}{m^{2+\kappa}} e^{- \frac{\decayrate}{m} (1+\kappa)(\tau(x)-\tauzero)},
\end{equation}
for any $\kappa > 0$. In addition, if $x \in \Mrn$ satisfies $\tau(x) \geq \tauzero$ and $r \geq R$ then we have
\begin{equation}
    \int_{\mathcal{P}_x} \left| p^v \right|^a \left| p^u \right|^b \left| \pslash \right|_{\gslash}^{c} f \, d \mu_x \leq C^{2 + a + b + c} \| f_0 \|_{L^\infty} \psuppconst^{2 + a + b + c}
    \frac{\rsuppconst^{2(1+b) + c}}{r^{2(1+ b) + c}} e^{-\frac{\decayrate}{m} (\tau(x) - \tauzero)},
\end{equation}
where we used double null coordinates and their conjugate coordinates on $\mathcal{P}$ to express the weight, $R > 3m$ denotes the constant chosen in Definition~\ref{defn_timecoord_Schw} and $a,b,c \geq 0$.
\end{thm}

\begin{rem}
After a finite time, the `untrapped' parts of the solution have either escaped to future null infinity or fallen into the black hole, so that we are left to deal with the `almost trapped' parts of the solution, which present main difficulty in showing decay. We also remark that the dimensionless constants $C, C_0, \decayrate$ can be quantified from the proof.
\end{rem}

Next we turn to the extremal Reissner--Nordstr\"om spacetime. We provide a precise version of Theorem~\ref{maintheoremERN} in the form of Theorems~\ref{maintheoremERNprecise} and~\ref{ERN_slowdecayprop}. The following Theorem~\ref{maintheoremERNprecise} shows that moments of solutions to the massless Vlasov equation decay at least at a polynomial rate, while Theorem~\ref{ERN_slowdecayprop} shows the sharpness of the rate of decay.

\begin{thm}[Polynomial decay on extremal Reissner--Nordstr\"om] \label{maintheoremERNprecise}
Assume $f_0 \in L^\infty(\mathcal{P}_0)$ satisfies Assumption~\ref{assumption_support} and let $f$ be the unique solution to the massless Vlasov equation on the extremal Reissner--Nordstr\"om background such that $f|_{\mathcal{P}_0} = f_0$. Let $w: \mathcal{P} \rightarrow [0,\infty)$ be smooth and bounded in $x$ according to Definition~\ref{boundedness_in_x}. Then Lemma~\ref{prop_compactness_support_general} applies and in particular $W < \infty$.

There exist dimensionless constants $C,C_0>0$ such that if $\tauzero = C_0 \left( m + \frac{\rsuppconst^2}{m} \right)$, then for times $\tau(x) \geq \tauzero$ the moment associated to $w$ decays at least at a quadratic rate:
\begin{equation}
		\int_{\mathcal{P}_x} w f \, d \mu_x \leq C W \| f_0 \|_{L^\infty} \frac{\psuppconst^2 \rsuppconst^2}{r^2} \frac{m^2}{(\tau(x)-\tauzero)^2}.
\end{equation}

Moreover, certain weights decay at a faster rate, depending on the distance from the event horizon. If we assume $r \geq (1+ \delta) m$ for some $0< \delta < 1$ and let $a,b,c \geq 0$, then the following specific weights expressed in $(t^*,r)$-coordinates decay at a faster rate:
\begin{equation}
    \int_{\mathcal{P}_x} \left| p^r \right|^a (p^{t^*})^b \left| \pslash \right|_{\gslash}^c f \, d \mu_x \leq \frac{C}{\delta^2} \| f_0 \|_{L^\infty} \frac{\psuppconst^{2+a+b+c} \rsuppconst^{2+c}}{r^{2+c}} \left( \frac{m}{\tau(x) - \tauzero} \right)^{2(1+a+b)+c},
\end{equation}
whereas if we assume that $m \leq r \leq (1+\delta) m$ then we obtain the bound
\begin{equation} \label{ERN_momentsthm_inequ}
    \int_{\mathcal{P}_x} \left| p^r \right|^a (p^{t^*})^b \left| \pslash \right|_{\gslash}^c f \, d \mu_x \leq C \| f_0 \|_{L^\infty} \frac{\psuppconst^{2+a+b+c} \rsuppconst^{2+c}}{r^{2+c}} \left( \frac{m}{\tau(x) - \tauzero} \right)^{2(1+a)+c}.
\end{equation}

If we assume that $x \in \Mern$ is such that $\tau(x) \geq \tauzero$ and $r \geq R$ then the following weights expressed in double null coordinates decay at the faster rate of
\begin{equation}
    \int_{\mathcal{P}_x} \left| p^v \right|^a \left| p^u \right|^b \left| \pslash \right|_{\gslash}^c f \, d \mu_x \leq C \| f_0 \|_{L^\infty} \frac{\psuppconst^{2+a+b+c} \rsuppconst^{2(1+b)+c}}{r^{2(1+b)+c}} \left( \frac{m}{\tau(x) - \tauzero} \right)^{2(1+a+b)+c}.
\end{equation}

If the initial distribution is supported away from the event horizon, we recover an exponential rate of decay. More precisely, let $0 < \delta < 1$ and assume 
\begin{equation}
    \supp(f_0) \subset \{ (x,p) \in \mathcal{P} : (1+\delta)m \leq r \}.
\end{equation}
Then there exists a constant $\decayrate>0$ independent of $m, f_0, w$ and $x$ such that we have the improved decay rate
\begin{equation}
    \int_{\mathcal{P}_x} w f \, d \mu_x \leq C W \| f_0 \|_{L^\infty} \frac{\rsuppconst^2 \psuppconst^2}{r^2} e^{-\frac{\decayrate}{m} \left( \tau(x) - \tauzero \right)}.
\end{equation}
In addition the same refinements about improving the rate of decay away from the photon sphere apply analogously to the Schwarzschild case.
\end{thm}

\begin{thm}[Sharpness of polynomial decay on extremal Reissner--Nordstr\"om] \label{ERN_slowdecayprop}
Under the same assumptions as in Theorem~\ref{maintheoremERNprecise}, there exist dimensionless constants $C, \constbsl > 0$ such that for all $0 < \delta < \frac{1}{2}$ there exists $\tauslow \sim \delta^{-1}$ such that for all $x \in \mathcal{H}^+$ with $\tau(x) > \tauslow$
\begin{equation} \label{ERN_lowerbound}
    \int_{\mathcal{P}_x} \left| p^r \right|^a \left| \pslash \right|_{\gslash}^b (p^{t^*})^c f \, d \mu_x \geq C \psuppconst^{2+a+b+c} \frac{m^{2(a+1) + b}}{\tau(x)^{2(a+1)+b}} \left( \inf_{(x,p) \in \badsetaplarge_\delta} \left| f_0(x,p) \right| \right),
\end{equation}
where we have abbreviated $\badsetaplarge_\delta = \badsetaplarge_{\constbsl,\delta}$ and assumed $a,b,c \geq 0$.
\end{thm}

By combining Theorems~\ref{maintheoremERNprecise} and~\ref{ERN_slowdecayprop} we immediately arrive at the conclusion that different components of the energy momentum tensor decay at different rates along the event horizon.

\begin{cor}
Assume $f_0 \in L^\infty(\mathcal{P}_0)$ satisfies Assumption~\ref{assumption_support} and let $f$ be the unique solution to the massless Vlasov equation on the extremal Reissner--Nordstr\"om background such that $f|_{\mathcal{P}_0} = f_0$. Then for all $0 < \delta < \frac{1}{2}$ and all $x \in \mathcal{H}^+$ with $\tau(x) > \max(\tauzero,\tauslow(\delta))$ we have
\begin{equation}
\begin{gathered}
    \psuppconst^4 \left( \inf\nolimits_{\badsetaplarge_\delta} \left| f_0 \right| \right) \frac{m^2}{\tau(x)^2} \lesssim T^{t^* t^*} \lesssim \frac{\psuppconst^4 \rsuppconst^2}{m^2} \| f_0 \|_{L^\infty} \frac{m^2}{(\tau(x)-\tauzero)^2}, \\
    \psuppconst^4 \left( \inf\nolimits_{\badsetaplarge_\delta} \left| f_0 \right| \right) \frac{m^4}{\tau(x)^4} \lesssim \left| T^{t^* r} \right| \lesssim \frac{\psuppconst^4 \rsuppconst^2}{m^2} \| f_0 \|_{L^\infty} \frac{m^4}{(\tau(x)-\tauzero)^4}, \\
    \psuppconst^4 \left( \inf\nolimits_{\badsetaplarge_\delta} \left| f_0 \right| \right) \frac{m^6}{\tau(x)^6} \lesssim T^{r r} \lesssim \frac{\psuppconst^4 \rsuppconst^2}{m^2} \| f_0 \|_{L^\infty} \frac{m^6}{(\tau(x)-\tauzero)^6}.
\end{gathered}
\end{equation}
\end{cor}

The following Theorem~\ref{ERN_nondecaytransversal} establishes that transversal derivatives of the energy mometum tensor do not decay along the extremal event horizon. Theorem~\ref{ERN_nondecaytransversal} is the precise version of Theorem~\ref{maintheorem_ern_nondecay_rough} above. Recall also Remark~\ref{rem_intro_nondecaythm}.

\begin{thm}[Non-decay of transversal derivatives on extremal Reissner--Nordstr\"om] \label{ERN_nondecaytransversal}
Under the same assumptions and using the same notation as in Theorem~\ref{maintheoremERNprecise}, let $0 < \delta < \frac{1}{2}$ and denote $\badsetaplarge_\delta = \badsetaplarge_{\constbsl,\delta}$. Assume in addition
\begin{itemize}
	\item $f_0 \in C^{\infty}(\mathcal{P}_0)$ and $f_0 \geq 0$,
	\item $\partial_{t^*} f |_{\Sigma_0} (x,p) \neq 0$ for all $(x,p) \in \badsetaplarge_\delta$,
	\item $\inf\nolimits_{\badsetaplarge_\delta} f_0 > 0$ and $\inf\nolimits_{\badsetaplarge_\delta} \left| \partial_{t^*} f_0 \right| > 0$.
\end{itemize}
Then there exist dimensionless constants $C,C_0 > 0$ independent of $\delta$ such that for all $x \in \mathcal{H}^+$ with $\tau(x) \geq C_0 \left( \tauzero + m \delta^{-1} + m \left( \inf\nolimits_{\badsetaplarge} \left| \partial_{t^*} f_0 \right| \right)^{-\frac{1}{2}} \right)$,
\begin{equation}
    \left| \partial_r \int_{\sphere} T^{t^* t^*}[f] \, d \omega \right| \geq C \psuppconst^4 \left( \inf\nolimits_{\badsetaplarge_\delta} \left| \partial_{t^*} f_0 \right| \right).
\end{equation}
In particular the transversal derivative of the energy momentum tensor does not decay along the event horizon.
\end{thm}
