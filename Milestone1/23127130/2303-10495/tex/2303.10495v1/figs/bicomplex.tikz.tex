\begin{tikzpicture}[scale=2.0,>=latex']
    \def\depth{2}
      
    % node placement and Delta-loops
    \foreach \n in {0,...,\depth} {
        \foreach \k in {0,...,\n} {
            \pgfmathtruncatemacro{\nk}{\n-\k}
            \node (\k_\nk) at (2*\k-\n,-3*\n/5) {\large\contour{white}{$\cC_{\k,\nk}$}};
            \draw (\k_\nk) edge [->,looseness=8,loop,out=65,in=115] node [anchor=center] {\contour{white}{$\Delta_Y$}} (\k_\nk);
            \draw (\k_\nk) edge [->,looseness=8,loop,out=245,in=295] node [anchor=center] {\contour{white}{$\Delta_X$}} (\k_\nk);
        }
    }

    % arrow drawing
    \foreach \n in {0,...,\depth} {

        \foreach \k in {0,...,\n} {
            \pgfmathtruncatemacro{\nk}{\n-\k}
            \pgfmathtruncatemacro{\nkdec}{\n-\k-1}
            \pgfmathtruncatemacro{\kdec}{\k-1}

            % draw arrows up and to the right
            \ifnum \nk > 0
                \ifodd\k
                    \draw (\k_\nk) edge [red,->] node [anchor=center] {\contour{white}{$-\del_Y$}} (\k_\nkdec);
                \else
                    \draw (\k_\nk) edge [->] node [anchor=center] {\contour{white}{$\del_Y$}} (\k_\nkdec);
                \fi
            \fi
          
            % draw arrows up and to the left
            \ifnum \k > 0
                \draw (\k_\nk) edge [->] node [anchor=center] {\contour{white}{$\del_X$}} (\kdec_\nk);
            \fi
        }
    }
\end{tikzpicture}
