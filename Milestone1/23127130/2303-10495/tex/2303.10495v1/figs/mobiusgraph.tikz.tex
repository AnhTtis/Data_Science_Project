\begin{tikzpicture}[]
\begin{groupplot}[
group style={group size=2 by 1, group name=myplots},
hide axis,
view={-25}{45},
zmin=-0.2, zmax=0.2,
samples=100,
samples y=0,
]
    % left - Mobius band
    \nextgroupplot
    \foreach \yval in {-0.1,-0.05,0.0,0.05,0.1} {
        \addplot3
        [domain=0:360]
        (%
        {(1.0+(\yval)*cos(x/2))*cos(x)},
        {(1.0+(\yval)*cos(x/2))*sin(x)},
        {(\yval)*sin(x/2)}
        );
    }

    \foreach \theta in {0,60,120,180,240,300} {
        \addplot3
        [mark=*,color=red]
        coordinates
        {%
        ({(1.0+0.1*cos(\theta/2))*cos(\theta)},{(1.0+0.1*cos(\theta/2))*sin(\theta)},{0.1*sin(\theta/2)})
        ({(1.0+0.05*cos(\theta/2))*cos(\theta)},{(1.0+0.05*cos(\theta/2))*sin(\theta)},{0.05*sin(\theta/2)})
        ({cos(\theta)},{sin(\theta)},{0.0})
        ({(1.0-0.05*cos(\theta/2))*cos(\theta)},{(1.0-0.05*cos(\theta/2))*sin(\theta)},{-0.05*sin(\theta/2)})
        ({(1.0-0.1*cos(\theta/2))*cos(\theta)},{(1.0-0.1*cos(\theta/2))*sin(\theta)},{-0.1*sin(\theta/2)})
        };
    }
    
    % right - base graph
    \nextgroupplot
    \addplot3
    [domain=0:360]
    (%
    {cos(x)},
    {sin(x)},
    {0.0}
    );

    \foreach \theta in {0,60,120,180,240,300} {
        \addplot3
        [mark=*,color=red]
        coordinates
        {%
        ({cos(\theta)},{sin(\theta)},{0.0})
        };
    }
\end{groupplot}

\draw (myplots c1r1.east) edge [->] node [anchor=center] {\contour{white}{\Large$\pi$}} (myplots c2r1.west);

\end{tikzpicture}
