% Table visualization bars:
\newcommand*{\minWidth}{35}
\newcommand*{\maxValue}{1}
\newcommand{\makeBar}[4]{%
    \hspace{-6.5pt}
  \tikz[baseline]{
    \node[anchor=base,text width=\minWidth,align=#2,inner sep=0pt,inner xsep=\tabcolsep,outer sep=0pt] (n) {\strut{#1}};
    \begin{pgfonlayer}{background}
        {
       \ifdim #1pt > 0.0pt 
            \edef\color{#3} 
            \pgfmathparse{0.5 + abs(#1/(2*\maxValue))}
            \edef\contents{{\pgfmathresult}}
            \fill[font=\boldmath,color=\color] ($(n.north west)!{0.5}!(n.north east)$) rectangle ($(n.south west)!{{\contents}}!(n.south east)$);
      \else
            \edef\color{#4}
            \pgfmathparse{0.5 - abs(#1/(2*\maxValue))}
            \edef\contents{{\pgfmathresult}}
            \fill[font=\boldmath,color=\color] ($(n.north west)!{{\contents}}!(n.north east)$) rectangle ($(n.south west)!{0.5}!(n.south east)$);
      \fi
        }
    \end{pgfonlayer}
  }
    \hspace{-9.5pt}
}
