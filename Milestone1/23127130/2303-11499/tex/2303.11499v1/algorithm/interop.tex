\begin{algorithm}[t]
\begin{footnotesize}
%\label{algo:inter-op}
\SetInd{0em}{1em}
\caption{\begin{small}Determination Inter-operation reuse opportunities for every edge in the DAG of tensor operations\end{small}}\label{alg:inter-op}
    \For{node $\in$ NODES}
{   
node.numcast=0	\\
\For{edge $\in$ edges(node)}
{
    flag=0\\
    \If{edge.isTransitive == false}{
    node.numcast++\\
    \If{node.numcast > 1}
    {
        \textcolor{LimeGreen}{node.parallel\_multicast=true}
    }
 

}
   \If{node.dominance$\neq$contracted$\And$edge.isTransitive==false}
   {
edge.inter\_dataflow=\textcolor{blue}{pipelineable}
   }
\If{node.dominance==contracted | node.op$\neq$tensor\_mac}
   {
edge.inter\_dataflow=sequential
   }
   \If{edge.dest.dominance $\notin$ edge.tensor.ranks}
   {
edge.inter\_dataflow=sequential
   }
   \If{node.dominance$\neq$contracted$\And$edge.isTransitive==true}
   {
\For{pathnode $\in$ longestpath(edge)}
{
\If{pathnode.dominance==contracted}
{edge.inter\_dataflow=\textcolor{BrickRed}{pipeline\_with\_writeback} \\ 
flag=1 \\
break
}
}
\If{flag==0}
{edge.inter\_dataflow=\textcolor{cyan}{pipeline\_with\_hold}
}
   }
}
}

\end{footnotesize}

\end{algorithm}
