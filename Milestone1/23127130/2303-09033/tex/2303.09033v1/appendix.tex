\subsection{Gamma Right Tail Concentration}

\begin{lem}
\label{lem:gam_conc}
If $X \sim \text{Gamma}(\alpha,\beta)$, then for any $a \in [0,\alpha/\beta]$, 
\[
Pr\Big( X < a \Big) \leq \bigg(\frac{a\beta}{\alpha}\bigg)^{\alpha} \exp(\alpha - a\beta)
\]
\end{lem}

\begin{proof}
Note for any $t>0$, we have
\[
Pr(X < a) = Pr(-tX > -ta) = Pr\big(\exp(-tX) > \exp(-ta)\big) \leq 
 \frac{\E{\exp(-tX)}}{\exp(-ta)},\]

where the last inequality follows from Markov inequality. Now it can be shown that:
\begin{align*}
\E{\exp(-tX)} 
   & = \frac{\beta^\alpha}{\Gamma(\alpha)} \int_{x = 0}^\infty \exp(-tx)\exp(-\beta x) x^{\alpha - 1}dx 
	\\
   & = \frac{\beta^\alpha}{\Gamma(\alpha)} \int_{u = 0}^\infty \exp(-u)\exp(-\beta x) \bigg(\frac{u}{t+\beta}\bigg)^{\alpha - 1}\frac{1}{(t+\beta)}du 
   = \frac{\beta^\alpha}{(t+\beta)^\alpha}.
\end{align*}

Let us denote by $u(t)= \frac{\beta^\alpha}{(t+\beta)^\alpha}$ for any $t \in \R_+$. Thus we get:

\begin{align}
\label{eq:gamma_conc}
Pr(X < a) = Pr(-tX > -ta) = Pr\big(\exp(-tX) > \exp(-ta)\big) \leq 
 u(t){\exp(ta)}.
\end{align}


Now minimizing the right hand side, $u(t)$, over all possible choices of $t > 0$, we get:

\[
\frac{d}{dt}u(t) =  \frac{\beta^\alpha}{(t+\beta)^\alpha} \bigg( a - \frac{\alpha}{t+\beta} \bigg) = 0,
\] 

which implies $\bigg( a - \frac{\alpha}{t+\beta} \bigg) = 0$ or $t = \frac{\alpha}{a}-\beta$, which in turn implies we must have $a < \frac{\alpha}{\beta}$, since $t > 0$ by choice. 

The final result now follows replacing $t = \frac{\alpha}{a}-\beta$ in \cref{eq:gamma_conc}. 
%
\end{proof}
