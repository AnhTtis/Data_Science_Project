\documentclass[]{standalone}
\usepackage{tikz}
\usetikzlibrary{intersections}
\newcommand{\Flow}{\varphi}

\begin{document}

\begin{tikzpicture}[every node/.style={scale=1.3}]
    \tikzstyle axes=[thin, color=black];
    \tikzstyle flow=[thick, color=blue!70!black];
    \tikzstyle control=[thick, color=red!80!black];
    \tikzstyle controlcirc=[fill, color=red!80!black];


    % axes
    \def\axes{+(10, 0) node[anchor=north west] {$t$} -- +(0, 0) -- +(0, 5)};
    \def\caxbasept{++(0, -3.8)};
    \draw[name path=flowaxes][<->, style=axes] \axes;
    \draw[<->, style=axes] \caxbasept +(10, 0) node[anchor=north west] {$t$} -- +(0, 0) -- +(0, 3);
    
    % flow curve
    \draw[style=flow, name path=flow_curve] (0, 1) .. controls (2, 4)  and (5, 2) .. (9, 5);
    
    % draw control values
    \foreach \t / \u / \utext in {0/0.5/u_1, 2.5/2/u_2, 5/2.5/u_3, 7.5/1/u_4}
    {
        \draw[style=control] \caxbasept +(\t, \u) -- +(\t+2.5, \u) node[midway, anchor=south] {$\utext$};
        \draw[style=help lines] \caxbasept +(\t, 0) -- +(\t, 2.8);
        \draw[style=controlcirc] \caxbasept +(\t, \u) circle (1pt);
    };
    
    % draw flow points at t = k\Delta
    \foreach \i / \t / \ttext in {0/0/0, 1/2.5/\Delta, 2/5/2\Delta, 3/7.5/3\Delta}
    {
        \path[name path=tpath] (\t, 0) -- (\t, 5);
        \draw[name intersections={of=tpath and flow_curve, by={a}}, style=help lines] (\t, 0) node[anchor=north] {$\ttext$} -- (a);
        \draw[name intersections={of=tpath and flow_curve, by={a}},fill, style=flow] (a) circle (1pt) node[anchor=south east] {$x_{\i}$};
    };
    
    % draw the final/particular point
    \def\spt{8.5}
    \path[name path=ourpoint] (\spt, 0) -- (\spt, 5);
    \draw[name intersections={of=ourpoint and flow_curve, by={a}}, style=help lines] (\spt, 0) node[color=black, anchor=north] {$s$} -- (a);
    \draw[name intersections={of=ourpoint and flow_curve, by={a}},fill, style=flow] (a) circle (1pt) node[anchor=north west] {
        $x_{4} = \Flow(s, x_0, u)$};
    
    \def\arrowlvl{0.1}
    % draw the $\tau_i$s
    \foreach \i / \tst / \tend in {1/0/2.5, 2/2.5/5, 3/5/7.5, 4/7.5/\spt}
        \draw[<->, style=help lines] (\tst, \arrowlvl) -- (\tend, \arrowlvl) node[color=black, midway, anchor=south] {$\tau_{\i}\Delta$};
    
\end{tikzpicture}

\end{document}
