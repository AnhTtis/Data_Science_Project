\begin{itemize}
    \item \textbf{Variational Autoencoders (VAEs)}: \cite{DBLP:journals/corr/KingmaW13,DBLP:conf/icml/RezendeMW14}.
    \item \textbf{Discrete latent representations}:
        \begin{itemize}
            \item \textit{Concrete distributions and Gumble-softmax trick:} \cite{DBLP:conf/iclr/JangGP17, DBLP:conf/iclr/MaddisonMT17}.
            \item via vectorization of the latent space, \cite{DBLP:conf/nips/OordVK17}; and clustering, \cite{DBLP:conf/ijcai/JiangZTTZ17}.
        \end{itemize}
    \item \textbf{Learn abstract latent states of an unknown MDP}:
        \begin{itemize}
            \item Learn discrete states via GANs for planning purposes: \cite{DBLP:conf/nips/KurutachTYRA18}.
            \item Learn continuous abstract states for reinforcement learning: \cite{DBLP:conf/nips/FreemanHM19}.
            \item Learn discrete states via VAEs and finite sequences of observations for planning purposes (prioritized sweeping): \cite{DBLP:conf/icml/CorneilGB18}.
        \end{itemize}
    \item \textbf{Deep reinforcement learning with continuous states and actions via stochastic policies}: REINFORCE, \cite{DBLP:books/lib/SuttonB98}; soft actor-critic \cite{DBLP:conf/icml/HaarnojaZAL18}.
    \item \textbf{The posterior collapse problem}:
    \begin{itemize}
        \item Analysis of the posterior collapse problem, \cite{DBLP:conf/icml/AlemiPFDS018, DBLP:conf/iclr/LucasTGN19}.
        \item Cross-entropy regularization, \cite{DBLP:conf/corl/BurkeHR19, DBLP:journals/corr/abs-1910-09588}.
    \end{itemize}
