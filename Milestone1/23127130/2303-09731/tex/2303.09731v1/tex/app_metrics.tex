\section{Two Popular Metrics to Evaluate PC Difference}
\label{appendix:metrics}

    \nonident \textbf{Chamfer Distance \cite{chong2019generating}.}
    \begin{align}
    D_C(S,S')= \frac{1}{||S'||}\sum_{y\in S'}min_{x\in S}||x-y||_2^2 \label{equal:chamfer}
    \end{align}
    \nonident \textbf{Average Square L2 Distance of kNN.}
    \begin{align}
    D_k(S,S')= \frac{1}{||S'||}\sum_{y\in S'}(\frac{1}{k}\sum_{x\in \text{kNN}(S,y)}||x-y||_2^2) \label{equal:knn}
    \end{align}
    In these equations, $S,S'$ denote two specific point sets defined above, $k$ denotes the hyper-parameter of $\text{kNN}(\cdot,\cdot)$ (we set $k=10$ in the following experiment), and $\text{kNN}(S,y)$ represents the $k$ nearest neighbor points of the point $y$ in the set $S$.
    % a subset of point set $S$, which includes $k$ nearest points of point $y$.
