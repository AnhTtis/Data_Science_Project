
\begin{algorithm}[ht]
  \caption{Fake Object Elimination}
  \label{alg:detecting}
\begin{algorithmic}[1]
  \STATE {\bfseries Input:} Input PC $X_t$, input space $D$, threshold $\beta$.
  \STATE {\bfseries Input:} Local objectness predictor $O$, boundary value $B$.
  \STATE {\bfseries Input:} Detector's original prediction $Preds$.
  \STATE {\bfseries Output:} Detector's final prediction $\overline{Preds}$.
  \STATE Split $D$ into equal-sized pillars: $Pillars = \text{split}(D)$.
  \FOR{Pillar $p_j\in Pillars$}
  \FOR{Point $x_i\in X_t$}
  \STATE Initialize: $pc_j=\emptyset$.
  \IF{$x_i$ {\bfseries inside} $p_j$}
  \STATE Generate a $7$-dim feature vector $x_i'$ for $x_i$.
  \STATE $pc_j=pc_j\bigcup x_i'$.
  \ENDIF
  \ENDFOR
  \STATE Calculate the objectness score: $\widehat{obj_j}=O(pc_j)$.
  \ENDFOR
  \STATE Initialize: $\overline{Preds}=\emptyset$.
  \FOR{$pred_k\in Preds$}
  \STATE Initialize: $cnt_k = 0, sum_k = 0$.
  \FOR{Pillar $p_j\in Pillars$}
  \IF{$IoU_{2D}(pred_k,p_j)>\beta$}
  \STATE $sum_k+=\widehat{obj_j}$
  \STATE $cnt_k+=1$
  \ENDIF
  \ENDFOR
  \IF{$sum_k/ cnt_k>B$}
  \STATE $\overline{Preds}=\overline{Preds}\bigcup pred_k$.
  \ENDIF
  \ENDFOR
  \RETURN $\overline{Preds}$
\end{algorithmic}
\end{algorithm}
