    \vskip-\baselineskip
    \penalty10000%

      \normalbaselines
      \ifdim\linewidth=\columnwidth
      \else
        \parshape\@ne
        \@totalleftmargin\linewidth
      \fi
      \vss
      \noindent#1\par}%
      \penalty10000%
      \vskip-\baselineskip}%
      \penalty10000}
