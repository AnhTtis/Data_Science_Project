

\section{Miscellaneous computations}
\label{appendix: Miscellaneous computations}


We provide in this section some miscellaneous computations, which hold as long as \( v_j(s_j, y_j) = e^{- \frac{|y_j|^2}{2} },\ j=1, \dots, N \).



\subsection{Conservative quantities in dimension \( d=2 \)}

We give the explicit expressions for the conservative quantities involved in Lemma \ref{lemma: conserved quantities in HO}, in the two-dimensional case.

The \( \mathbb{L}^2 \) norm of a sum of \( N \) bubbles is given by
\begin{align*}
    \|u\|_{ \mathbb{L}^2 }^2 &= \sum_{k,l=1}^N \frac{A_k A_l}{L_k L_l} \langle b_{(d+2)(k-1)+1}, b_{(d+2)(l-1)+1} \rangle.
\end{align*}
%


The energy of a sum of bubbles is given by
\begin{align*}
    2E_0 &= \left\langle \sum_{j=1}^N H u_j(x), \sum_{k=1}^N u_k(x) \right\rangle \\
    &= \sum_{j, k=1}^N \left\langle H u_j(x), u_k(x) \right\rangle \\
    &= \sum_{j, k=1}^N \left( \left\langle \nabla_x u_j(x), \nabla_x u_k(x) \right\rangle + \left\langle |x|^2 u_j(x), u_k(x) \right\rangle \right)
\end{align*}
%
We have
\begin{align*}
    \nabla_x u_k(x) 
    &= \frac{A_k}{L_k} \frac{1}{L_k} \left[ iL_k\beta_k - \frac{2+iB_k}{2} \frac{x-X_k}{L_k} \right] e^{i\Gamma_k} e^{-\frac{1}{2} \left| \frac{x-X_k}{L_k} \right|^2} \\
    &= \frac{A_k}{L_k} \left[ i\beta_k - \frac{2+iB_k}{2L_k^2} (x-X_k) \right] e^{i\Gamma_k} e^{-\frac{1}{2} \left| \frac{x-X_k}{L_k} \right|^2} \\
    &= \frac{A_k}{L_k} \left[ \underbrace{i\beta_k + \frac{2+iB_k}{2L_k^2} X_k}_{=: \eta_k} - \frac{2 + iB_k}{2L_k^2} x \right] e^{i\Gamma_k} e^{-\frac{1}{2} \left| \frac{x-X_k}{L_k} \right|^2} \\
    &= \left[ \underbrace{i\beta_k + \frac{2+iB_k}{2L_k^2} X_k}_{=: \eta_k} - \frac{2 + iB_k}{2L_k^2} x \right] u_k(x)
,\end{align*}
%
and thus
\begin{align*}
    \left| \nabla_x u(x) \right|^2 &= \nabla_x u(x) \cdot \overline{\nabla_x u(x)}  \\
    &= \sum_{k, l=1}^N \left[ \eta_k - \frac{2 + iB_k}{2L_k^2} x \right] \cdot \left[ \overline{\eta_l} - \frac{2 - iB_l}{2L_l^2} x \right] u_k(x) \overline{u_l}(x) \\
    &= \sum_{k, l=1}^N \left[ \eta_k \cdot \overline{\eta_l} - \frac{2 + iB_k}{2L_k^2} x \cdot \overline{\eta_l} - \frac{2 - iB_l}{2L_l^2} x \cdot \eta_k + \frac{2 - iB_l}{2L_l^2} \frac{2 + iB_k}{2L_k^2} |x|^2 \right] u_k(x) \overline{u_l}(x) \\
    &= \sum_{k, l=1}^N \left[ \eta_k \cdot \overline{\eta_l} - \left( \frac{2 + iB_k}{2L_k^2} \overline{\eta_l} + \frac{2 - iB_l}{2L_l^2} \eta_k \right) \cdot x + \frac{2 - iB_l}{2L_l^2} \frac{2 + iB_k}{2L_k^2} |x|^2 \right] u_k(x) \overline{u_l}(x)
\end{align*}
%
Finally, to obtain \( \langle  \nabla_x u(x),  \nabla_x u(x) \rangle \) it suffices to integrate \( \left| \nabla_x u(x) \right|^2 \) over \( \mathbb{R}^d \). We then recognize integrals that have already been computed in Appendix \ref{appendix: fourier transforms of gaussians}.

We now proceed to computing the momentum, given by 
\begin{equation*}
    M_0 = -2E_0 \| x\psi\|^2_{ \mathbb{L}^2 } + \|xu\|^4_{ \mathbb{L}^2 } + \left( \Im \int x\cdot \nabla u(x)\overline{u}(x) dx \right)^2
.\end{equation*}
%
We know how to compute \( E_0 \) from the previous computations. The norm \( \| x\psi\|^2_{ \mathbb{L}^2 } \) has also already been computed in Appendix \ref{appendix: fourier transforms of gaussians}. Thus, it only remains to compute the integral
\begin{align*}
    \int x\cdot \nabla u(x)\overline{u}(x) dx
    &= \int_{ \mathbb{R}^d } \overline{u}(x) \sum_{k=1}^N x\cdot \left[ \eta_k - \frac{2 + iB_k}{2L_k^2} x \right] u_k(x) dx \\
    &= \sum_{k=1}^N \int_{ \mathbb{R}^d } \overline{u}(x) \left[ \eta_k \cdot x - \frac{2 + iB_k}{2L_k^2} |x|^2 \right] u_k(x) dx
.\end{align*}
%
Once again, this integral is computable thanks to the results of Appendix \ref{appendix: fourier transforms of gaussians}.
