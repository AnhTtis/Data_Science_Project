\newcommand{\createbutton}[2]{

    jsaction={
        var fld = this.addField("ToCopy", "text", \thepage-1, this.getPageBox({nPage: \thepage-1}));
        fld.value = "#2";
        fld.setFocus();
        app.alert("Copied to Clipboard!");
        app.execMenuItem("Copy");
        this.removeField("ToCopy");     
    }
]{\fbox{\strut #1}}
