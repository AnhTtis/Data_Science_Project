\newrobustcmd{\mth}[1]{%

% command for mass vertex
\newrobustcmd{\mass}[2][]{%
  {\fmfv{decoration.shape=circle,decoration.filled=20,label=#1,label.dist=8,label.angle=135}{#2}}
}%

% command for charge vertex

% command for scalar 
\newrobustcmd{\scalar}[2][]{%
  {\fmf{wiggly,tension=1.,label=#1,label.dist=5}{#2}}
}%

% command for scalar graviton 
\newrobustcmd{\graviton}[2][]{%
  {\fmf{dbl_wiggly,tension=1.,label=#1,label.dist=5}{#2}}
}%

% command for scalar graviton

% command for scalar graviton

}%

% command for some diagram

	\fmfstraight%

      \end{fmfgraph*}%
    \end{fmffile}%
    \vspace{2pt}%
    \vfill%
    %\captionof*{figure}{}%
  \end{minipage}}%
