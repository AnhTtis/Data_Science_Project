\begin{proof}
    By simultaneous induction over the following claims: \begin{enumerate}
        \item \label{lem:tight-nfs:1} If $\Phi \tr \seqi{\Gam}{t}{\stype \ra \tneutral \tim \stype'}{(0,0,s)}$, then $t \in \neutral$.
        \item \label{lem:tight-nfs:2} If $\Phi \tr \seqi{\Gam}{t}{\stype \ra \tightt \tim \stype'}{(0,0,s)}$, then $t \in \normal$.
    \end{enumerate}
    \begin{enumerate}
        \item By induction over $\Phi$: \begin{itemize}
            \item Case $\Phi$ ends with ($\ruleApp$). Then $t = v u$, $\Phi_u \tr \seqi{\Gam_u}{u}{\stype \ra \M \tim \stype'}{(0,0,r_u)}$, $\Phi_v \tr \seqi{\Gam_v}{v}{\M \ra \stype' \ra \del \tim \stype''}{(0,0,0,r_v)}$, such that $\Gam_t = \Gam_u + \Gam_v$ tight, $\del = \tneutral$, and $r = r_u + r_v$. But, by~\cref{lem:tight-zero}, $\stype \ra \M \tim \stype'$ tight. Absurd, since $\M \not\in \tightt$. Therefore, this case does not apply.
            \item Case $\Phi$ ends with ($\ruleAppPOne$). This case does not apply since $\tneutral \not= $.
            \item Case $\Phi$ ends with ($\ruleAppPTwo$). This case does not apply since $\tneutral \not= $.
            \item Case $\Phi$ ends with ($\ruleGet$). This case does not apply since $g > 0$.
            \item Case $\Phi$ ends with ($\ruleSet$). This case does not apply since $s > 0$.
        \end{itemize}
        \item By induction over $\Phi$: \begin{itemize}
            \item Case $\Phi$ ends with ($\ruleApp$). Then $t = v u$, $\Phi_u \tr \seqi{\Gam_u}{u}{\stype \ra \M \tim \stype'}{(0,0,0,r_u)}$, $\Phi_v \tr \seqi{\Gam_v}{v}{\M \ra \stype' \ra \del \tim \stype''}{(0,0,r_v)}$, such that $\Gam_t = \Gam_u + \Gam_v$ tight, $\del = $, and $r = r_u + r_v$. But, by~\cref{lem:tight-zero}, $\stype \ra \M \tim \stype'$ tight. Absurd, since $\M \not\in \tightt$. Therefore, this case does not apply.
            \item Case $\Phi$ ends with ($\ruleAppPOne$). Then $t = v u$, $\Phi_u \tr \seqi{\Gam_u}{u}{\stype \ra \tneutral \tim \stype'}{(0,0,r_u)}$, $\Phi_v \tr \seqi{\Gam_v}{v}{\tightv}{(0,0,0,r_v)}$, such that $\Gam_t = \Gam_u + \Gam_v$ tight and $r = r_u + r_v$. By the \ih (\cref{lem:tight-nfs:2}), we have $u \in \neutral$. Therefore, $v u \in \neutral$.
            \item Case $\Phi$ ends with ($\ruleAppPTwo$). Then $t = v u$, $\Phi_u \tr \seqi{\Gam_u}{u}{\stype \ra \tightt \tim \stype'}{(0,0,r_u)}$, $\Phi_v \tr \seqi{\Gam_v}{v}{\tvar}{(0,0,r_v)}$, such that $\Gam_t = \Gam_u + \Gam_v$ tight and $r = r_u + r_v$. By the \ih(\cref{lem:tight-nfs:3}), we have $u \in \normal$. By~\cref{prop:tightvalue}, we have $v = x$. Therefore, $x u \in \neutral$.
            \item Case $\Phi$ ends with ($\ruleGet$). This case does not apply since $g > 0$.
            \item Case $\Phi$ ends with ($\ruleSet$). This case does not apply since $s > 0$.
        \end{itemize}
        \item This case is subsumed by the two previous cases.
    \end{enumerate}
\end{proof}
