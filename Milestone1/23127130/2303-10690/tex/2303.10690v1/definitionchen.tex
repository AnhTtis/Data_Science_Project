% for number using a comma as a thousands separator
\newcount\ppnum
\newcommand\ppnumber[1]{%
	\ppnum=#1\relax
	\ifnum\ppnum<0
	$-$%
	\ppnum=-\ppnum
	\fi
	\let\pptemp\empty
	\loop\ifnum\ppnum>999
	\count255=\ppnum
	\divide\ppnum by1000
	\count255=\numexpr \count255 - 1000*\ppnum \relax
	\edef\pptemp{,\ifnum\count255<100 0\ifnum\count255<10 0\fi\fi
		\the\count255 \pptemp}%
	\repeat
	\the\ppnum
	\pptemp
}
