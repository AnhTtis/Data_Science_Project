To bound the cluster deviation error, we recursively expand of~\eqref{eq:clu_err1} and yield: 
\begin{align} \label{eq:clu_err2}
    &\sum\limits_{j\in\mathcal S_c}\rho_{j,c}\mathbb E\big\Vert\mathbf w_j^{(t)} - \bar{\mathbf w}_c^{(t)}\big\Vert
    \nonumber \\&
    \leq \chi_c^{(t)}\Big[(1+2\eta_{t-1}\beta)\sum\limits_{j\in\mathcal S_c}\rho_{j,c}\mathbb E\Vert{\mathbf{w}}_j^{(t-1)}-\bar{\mathbf w}_c^{(t-1)}\Vert
    +\eta_{t-1}(2\sigma+\delta_c)\Big]
    \nonumber \\&
    \leq 
    \prod_{m=t_{k-1}}^{t-1}\chi_c^{(m+1)}(1+2\eta_m\beta)\underbrace{\sum\limits_{j\in\mathcal S_c}\rho_{j,c}\mathbb E\Vert{\mathbf{w}}_j^{(t_{k-1})}-\bar{\mathbf w}_c^{(t_{k-1})}\Vert}_{(i)}
    \nonumber \\&
    +\sum_{\ell=t_{k-1}}^{t-1}\chi_c^{(\ell+1)}\eta_\ell\prod_{j=\ell+1}^{t-1}\chi_c^{(j+1)}(1+2\eta_j\beta)\left(\delta_c+2 \sigma\right).
\end{align}
Considering (i) on the right hand side of~\eqref{eq:clu_err2}, we use the definition of $\mathbf w_i^{(t)}$ at $t=t_{k-1}$ from~\eqref{eq:aggr_alpha} and~\eqref{eq:w_i-gen} to obtain:
\begin{align} \label{eq:alpha1}
    \mathbf w_i^{(t_{k-1})} =& 
    \alpha_{k-1}\bar{\mathbf{w}}^{(t_{k-1}-\Delta)}
    +(1-\alpha_{k-1})\left[\chi_c^{(t_{k-1})}\widetilde{\mathbf{w}}_i^{(t_{k-1})}+(1-\chi_c^{(t_{k-1})})\sum\limits_{j\in\mathcal S_c}\rho_{j,c}\widetilde{\mathbf{w}}_j^{(t_{k-1})}\right].
\end{align}
In addition, using the definition of $\bar{\mathbf w}_c^{(t)}$, we obtain:
\begin{align} \label{eq:alpha_c1}
\bar{\mathbf w}_c^{(t_{k-1})} =& 
    \sum\limits_{j\in\mathcal S_c}\rho_{j,c}\mathbf w_j^{(t_{k-1})}
    \nonumber \\
    =&\alpha_{k-1}\bar{\mathbf{w}}^{(t_{k-1}-\Delta)}
    +(1-\alpha_{k-1})\sum\limits_{j\in\mathcal S_c}\rho_{j,c}\left[\chi_c^{(t_{k-1})}\widetilde{\mathbf{w}}_j^{(t_{k-1})}+(1-\chi_c^{(t_{k-1})})\sum\limits_{j\in\mathcal S_c}\rho_{j,c}\widetilde{\mathbf{w}}_j^{(t_{k-1})}\right]
    \nonumber \\
    =&\alpha_{k-1}\bar{\mathbf{w}}^{(t_{k-1}-\Delta)}
    +(1-\alpha_{k-1})\left[\chi_c^{(t_{k-1})}\sum\limits_{j\in\mathcal S_c}\rho_{j,c}\widetilde{\mathbf{w}}_j^{(t_{k-1})}+(1-\chi_c^{(t_{k-1})})\sum\limits_{j\in\mathcal S_c}\rho_{j,c}\widetilde{\mathbf{w}}_j^{(t_{k-1})}\right].
\end{align}

Taking the norm-2 from the both hand sides of the above equality yields:
\begin{align}
    \big\Vert\mathbf w_i^{(t_{k-1})}-\bar{\mathbf w}_c^{(t_{k-1})}\big\Vert
    &= \big\Vert(1-\alpha)\chi_c^{(t_{k-1})}[\widetilde{\mathbf{w}}_i^{(t_{k-1})}-\bar{\mathbf w}_c^{(t_k)}]\big\Vert
    \nonumber \\&
    =(1-\alpha)\chi_c^{(t_k)}\big\Vert\widetilde{\mathbf{w}}_i^{(t_k)}-\bar{\mathbf w}_c^{(t_k)}\big\Vert
    \nonumber \\&
    =(1-\alpha)\chi_c^{(t_k)}\big\Vert\widetilde{\mathbf{w}}_i^{(t_k)}-\sum\limits_{j\in\mathcal S_c}\rho_{j,c}\widetilde{\mathbf{w}}_j^{(t_k)}\big\Vert
    \nonumber \\&
    =(1-\alpha)\chi_c^{(t_k)}\big\Vert\sum\limits_{j\in\mathcal S_c}\rho_{j,c}\big(\widetilde{\mathbf{w}}_i^{(t_k)}-\widetilde{\mathbf{w}}_j^{(t_k)}\big)\big\Vert
    \nonumber \\&
      \overset{(a)}{\leq}
      (1-\alpha)\chi_c^{(t_k)}\sum\limits_{j\in\mathcal S_c}\rho_{j,c}\left\Vert \widetilde{\mathbf{w}}^{(t)}_{{i}}-\widetilde{\mathbf{w}}^{(t)}_{{j}}\right\Vert,~ i\in\mathcal S_c,
\end{align}
where (a) comes from Jensen's inequality. Applying Definition~\ref{def:clustDiv} into the above inequality gives us:
\begin{align} \label{eq:e_temp}
    &\Vert\mathbf e_{i}^{(t_k)}\Vert
      \leq
      (1-\alpha)\chi_c^{(t_k)}\sum\limits_{j\in\mathcal S_c}\rho_{j,c}\left\Vert \widetilde{\mathbf{w}}^{(t_k)}_{{i}}-\widetilde{\mathbf{w}}^{(t_k)}_{{j}}\right\Vert
      \leq
      (1-\alpha)\chi_c^{(t_k)}\Upsilon^{(t_k)}_{{c}},~ i\in\mathcal S_c.
\end{align}
Combining Lemma~\ref{lem:consensus-error} and~\eqref{eq:e_temp}, we obtain
\begin{align}
    &\Vert\mathbf e_{i}^{(t)}\Vert
      \leq
      (1-\alpha^{(t)})\chi_c^{(t)}\Upsilon^{(t)}_{{c}},~ i\in\mathcal S_c.
\end{align}
and conclude the proof. 

Finally, applying Definition~\ref{def:clustDiv} into the above inequality gives us:
\begin{align} \label{eq:epsilon_use}
    &\Vert\mathbf e_{i}^{(t)}\Vert
      \leq
      \chi_c^{(t)}\sum\limits_{j\in\mathcal S_c}\rho_{j,c}\left\Vert \widetilde{\mathbf{w}}^{(t)}_{{i}}-\widetilde{\mathbf{w}}^{(t)}_{{j}}\right\Vert
      \leq
      \chi_c^{(t)}\Upsilon^{(t)}_{{c}},~ i\in\mathcal S_c.
\end{align}
