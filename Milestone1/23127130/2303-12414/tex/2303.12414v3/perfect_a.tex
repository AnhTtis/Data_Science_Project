\begin{proposition} The difference between the perfect global model without delay (perfect aggregation) and with delay (weighted by $\alpha_k$) is upper-bounded as:
    \begin{align}
        \sum\limits_{c=1}^N \varrho_c\sum\limits_{j\in\mathcal S_c}\rho_{j,c}\mathbb E\Vert\mathbf w_j^{(t_k)} - \widehat{\mathbf w}^{(t_k)}\Vert\leq\phi^{(t_k)},
    \end{align}
    where $\widehat{\mathbf w}^{(t_k)}=\sum_{c=1}^N \varrho_{c}\sum_{j\in\mathcal{S}_c} \rho_{j,c} \tilde{\mathbf w}_j^{(t_k)}$ represents the perfect global model obtained from perfect synchronization, i.e., $\Delta=0$, at $t=t_k$ and 
    \begin{align}
        \phi^{(t_k)}&=
        (1-\alpha_k)\sum\limits_{c=1}^N \varrho_c\sum\limits_{j\in\mathcal S_c}\rho_{j,c}\mathbb E\left\Vert\mathbf w_j^{(t_{k-1})}-\widehat{\mathbf w}^{(t_{k-1})}\right\Vert
        % \nonumber \\&
        +\alpha_k\beta\sum\limits_{m=t_k-1}^{t_k-\Delta}\eta_m\sum\limits_{c=1}^N \varrho_c\sum\limits_{j\in\mathcal S_c}\rho_{j,c}\Vert\mathbf w_j^{(m)}-\mathbf w_j^*\Vert
        \nonumber \\&
        +2(1-\alpha_k) \beta\sum\limits_{m=t_k-1}^{t_{k-1}}\eta_{m}\sum\limits_{c=1}^N \varrho_c\sum\limits_{j\in\mathcal S_c}\rho_{j,c}\Vert\mathbf w_j^{(m)}-\mathbf w_j^*\Vert
        % \nonumber \\&
        +\alpha_k\sum\limits_{m=t_k-1}^{t_k-\Delta}\eta_m\sigma
        +2(1-\alpha_k) \sum\limits_{m=t_k-1}^{t_{k-1}}\eta_{m}\sigma.
    \end{align}
\end{proposition}
Using the definition of $\mathbf w_i^{(t)}$ at $t=t_k$ from~\eqref{eq:aggr_alpha} and obtain:
\begin{align} \label{eq:alpha}
    \mathbf w_i^{(t_{k})} =& \alpha_k\bar{\mathbf{w}}^{(t_{k}-\Delta)}
    +(1-\alpha_k)\left[\chi_c^{(t_k)}\widetilde{\mathbf{w}}_i^{(t_k)}+(1-\chi_c^{(t_k)})\sum_{j\in\mathcal{S}_c} \rho_{j,c} \tilde{\mathbf w}_j^{(t_k)}\right]
    \nonumber \\&
    = 
    \alpha_k\bar{\mathbf{w}}^{(t_{k}-\Delta)}
    +(1-\alpha_k)\left[\chi_c^{(t_k)}[\mathbf w_i^{(t-1)}-\eta_{t-1} \widehat{\mathbf g}_{i}^{(t-1)}]+(1-\chi_c^{(t_k)})\sum_{j\in\mathcal{S}_c} \rho_{j,c} [\mathbf w_j^{(t-1)}-\eta_{t-1} \widehat{\mathbf g}_{j}^{(t-1)}]\right]
\end{align} 
Here we define 
\begin{align} \label{eq:perfect_sync}
    \widehat{\mathbf w}^{(t_k)}&=\sum_{c=1}^N \varrho_{c}\sum_{j\in\mathcal{S}_c} \rho_{j,c} \tilde{\mathbf w}_j^{(t_k)}
    \nonumber \\&
    =\sum_{c=1}^N \varrho_{c}\sum_{j\in\mathcal{S}_c} \rho_{j,c} [\mathbf w_j^{(t-1)}-\eta_{t-1} \widehat{\mathbf g}_{j}^{(t-1)}]
    \nonumber \\&
    =\widehat{\mathbf w}^{(t_k)} -\eta_{t-1} \sum_{c=1}^N \varrho_{c}\sum_{j\in\mathcal{S}_c} \rho_{j,c}\widehat{\mathbf g}_{j}^{(t-1)}
\end{align}
as the model obtained from perfect synchronization, i.e., $\Delta=0$, at $t=t_k$.
Subtracting~\eqref{eq:alpha} with~\eqref{eq:perfect_sync} yields:
\begin{align}
    &\mathbf w_i^{(t_k)} - \widehat{\mathbf w}^{(t_k)} 
    = \alpha_k\bar{\mathbf{w}}^{(t_{k}-\Delta)}
    \nonumber \\&
    +(1-\alpha_k)\left[\chi_c^{(t_k)}[\mathbf w_i^{(t-1)}-\eta_{t-1} \widehat{\mathbf g}_{i}^{(t-1)}]+(1-\chi_c^{(t_k)})\sum_{j\in\mathcal{S}_c} \rho_{j,c} [\mathbf w_j^{(t-1)}-\eta_{t-1} \widehat{\mathbf g}_{j}^{(t-1)}]\right]-\widehat{\mathbf w}^{(t_k)}
    \nonumber \\&
    = \alpha_k\big[\bar{\mathbf{w}}^{(t_{k}-\Delta)}-\widehat{\mathbf w}^{(t_k)}\big]
    \nonumber \\&
    +(1-\alpha_k)\left[\chi_c^{(t_k)}[\mathbf w_i^{(t_k-1)} - \eta_{t_{k}-1}\widehat{\mathbf g}_{i}^{(t_k-1)}-\widehat{\mathbf w}^{(t_k)}]+(1-\chi_c^{(t_k)})\Big[\sum_{j\in\mathcal{S}_c} \rho_{j,c} [\mathbf w_i^{(t_k-1)} - \eta_{t_{k}-1}\widehat{\mathbf g}_{i}^{(t_k-1)}]-\widehat{\mathbf w}^{(t_k)}\Big]\right]
    \nonumber \\&
    = \alpha_k\sum\limits_{m=t_k-1}^{t_k-\Delta}\eta_m\sum\limits_{c=1}^N \varrho_c\sum\limits_{j\in\mathcal S_c}\rho_{j,c}\widehat{\mathbf g}_{j}^{(m)}
    \nonumber \\&
    +(1-\alpha_k)\bigg[\chi_c^{(t_k)}\Big[\mathbf w_i^{(t_k-1)}-\widehat{\mathbf w}^{(t_k-1)} - \eta_{t_{k}-1}[\widehat{\mathbf g}_{i}^{(t_k-1)}-\sum\limits_{c=1}^N \varrho_c\sum\limits_{j\in\mathcal S_c}\rho_{j,c}\widehat{\mathbf g}_{j}^{(t_k-1)}]\Big]
    \nonumber \\&
    +(1-\chi_c^{(t_k)})\Big[\sum_{j\in\mathcal{S}_c} \rho_{j,c}\mathbf w_j^{(t_k-1)}-\widehat{\mathbf w}^{(t_k-1)} - \eta_{t_{k}-1}[\sum_{j\in\mathcal{S}_c} \rho_{j,c}\widehat{\mathbf g}_{j}^{(t_k-1)}-\sum\limits_{c=1}^N \varrho_c\sum\limits_{j\in\mathcal S_c}\rho_{j,c}\widehat{\mathbf g}_{j}^{(t_k-1)}]\Big]\bigg]
    \nonumber \\&
    = \alpha_k\sum\limits_{m=t_k-1}^{t_k-\Delta}\eta_m\sum\limits_{c=1}^N \varrho_c\sum\limits_{j\in\mathcal S_c}\rho_{j,c}\nabla F_j(\mathbf w_j^{(m)}) 
    \nonumber \\&
    +\alpha_k\sum\limits_{m=t_k-1}^{t_k-\Delta}\eta_m\sum\limits_{c=1}^N \varrho_c\sum\limits_{j\in\mathcal S_c}\rho_{j,c}[\widehat{\mathbf g}_{j}^{(m)}-\nabla F_j(\mathbf w_j^{(m)})]
    \nonumber \\&
    +(1-\alpha_k)\chi_c^{(t_k)}\left[\mathbf w_i^{(t_k-1)}-\widehat{\mathbf w}^{(t_k-1)}\right]
    % \nonumber \\&
    +(1-\alpha_k)(1-\chi_c^{(t_k)})\left[\sum\limits_{j\in\mathcal S_c}\rho_{j,c}\mathbf w_j^{(t_k-1)}-\widehat{\mathbf w}^{(t_k-1)}\right]
    \nonumber \\&
    -(1-\alpha_k)\chi_c^{(t_k)} \left[\eta_{t_{k}-1}\big[\nabla F_i(\mathbf w_i^{(t_k-1)})+\widehat{\mathbf g}_{i}^{(t_k-1)}-\nabla F_i(\mathbf w_i^{(t_k-1)})\big]\right]
    \nonumber \\&
    -(1-\alpha_k)(1-\chi_c^{(t_k)})\left[\eta_{t_{k}-1}\sum\limits_{j\in\mathcal S_c}\rho_{j,c}\big[\nabla F_j(\mathbf w_j^{(t_k-1)})+\widehat{\mathbf g}_{j}^{(t_k-1)}-\nabla F_j(\mathbf w_j^{(t_k-1)})\big]\right]

Taking the norm-2 from the both hand sides of the above equality and applying the triangle inequality yields:
\begin{align} \label{eq:tri_e}

    \nonumber \\&

    \nonumber \\&
    +(1-\alpha_k)\chi_c^{(t_k)} \eta_{t_{k}-1}\big[\Vert\nabla F_i(\mathbf w_i^{(t_k-1)})\Vert+\Vert\widehat{\mathbf g}_{i}^{(t_k-1)}-\nabla F_i(\mathbf w_i^{(t_k-1)})\Vert\big]
    \nonumber \\&
    +(1-\alpha_k)(1-\chi_c^{(t_k)})\eta_{t_{k}-1}\Big\Vert\sum\limits_{j\in\mathcal S_c}\rho_{j,c}\big[\nabla F_j(\mathbf w_j^{(t_k-1)})+\widehat{\mathbf g}_{j}^{(t_k-1)}-\nabla F_j(\mathbf w_j^{(t_k-1)})\big]\Big\Vert
    \nonumber \\&
    +(1-\alpha_k) \eta_{t_{k}-1}\Big\Vert\sum\limits_{c=1}^N \varrho_c\sum\limits_{j\in\mathcal S_c}\rho_{j,c}\big[\nabla F_j(\mathbf w_j^{(t_k-1)})+\widehat{\mathbf g}_{j}^{(t_k-1)}-\nabla F_j(\mathbf w_j^{(t_k-1)})\big]\Big\Vert.
\end{align}
Applying Jensen's inequality on the above inequality, we obtain:
\begin{align}
    &\Vert\mathbf w_i^{(t_k)} - \widehat{\mathbf w}^{(t_k)}\Vert
    \leq

    \nonumber \\&

    % \nonumber \\&
    +(1-\alpha_k)(1-\chi_c^{(t_k)})\sum\limits_{j\in\mathcal S_c}\rho_{j,c}\Big\Vert\mathbf w_j^{(t_k-1)}-\widehat{\mathbf w}^{(t_k-1)}\Big\Vert
    \nonumber \\&
    +(1-\alpha_k)\chi_c^{(t_k)} \eta_{t_{k}-1}\Vert\nabla F_i(\mathbf w_i^{(t_k-1)})\Vert
    \nonumber \\&
    +(1-\alpha_k)\chi_c^{(t_k)} \eta_{t_{k}-1}\Vert\widehat{\mathbf g}_{i}^{(t_k-1)}-\nabla F_i(\mathbf w_i^{(t_k-1)})\Vert
    \nonumber \\&
    +(1-\alpha_k)(1-\chi_c^{(t_k)})\eta_{t_{k}-1}\sum\limits_{j\in\mathcal S_c}\rho_{j,c}\Big\Vert\big\nabla F_j(\mathbf w_j^{(t_k-1)})\Big\Vert
    \nonumber \\&
    +(1-\alpha_k)(1-\chi_c^{(t_k)})\eta_{t_{k}-1}\sum\limits_{j\in\mathcal S_c}\rho_{j,c}\Big\Vert\widehat{\mathbf g}_{j}^{(t_k-1)}-\nabla F_j(\mathbf w_j^{(t_k-1)})\Big\Vert

    +(1-\alpha_k)\sum\limits_{c=1}^N \varrho_c\sum\limits_{j\in\mathcal S_c}\rho_{j,c}\mathbb E\left\Vert\mathbf w_j^{(t_{k-1})}-\widehat{\mathbf w}^{(t_{k-1})}\right\Vert
    \nonumber \\&
    +2(1-\alpha_k) \sum\limits_{m=t_k-1}^{t_{k-1}}\eta_{m}\sum\limits_{c=1}^N \varrho_c\sum\limits_{j\in\mathcal S_c}\rho_{j,c}\Vert\nabla F_j(\mathbf w_j^{(m)})\Vert
    \nonumber \\&
    +2(1-\alpha_k) \sum\limits_{m=t_k-1}^{t_{k-1}}\eta_{m}\sum\limits_{c=1}^N \varrho_c\sum\limits_{j\in\mathcal S_c}\rho_{j,c}\sqrt{\mathbb E\Vert\widehat{\mathbf g}_{j}^{(m)}-\nabla F_i(\mathbf w_j^{(m)})\Vert^2}
    \nonumber \\&
    \overset{(b)}{\leq}
    \alpha_k\sum\limits_{m=t_k-1}^{t_k-\Delta}\eta_m\sum\limits_{c=1}^N \varrho_c\sum\limits_{j\in\mathcal S_c}\rho_{j,c}\Big\Vert\nabla F_j(\mathbf w_j^{(m)})\Big\Vert 
    \nonumber \\&
    +\alpha_k\sum\limits_{m=t_k-1}^{t_k-\Delta}\eta_m\sigma
    \nonumber \\&
    +(1-\alpha_k)\sum\limits_{c=1}^N \varrho_c\sum\limits_{j\in\mathcal S_c}\rho_{j,c}\mathbb E\left\Vert\mathbf w_j^{(t_{k-1})}-\widehat{\mathbf w}^{(t_{k-1})}\right\Vert
    \nonumber \\&
    +2(1-\alpha_k) \sum\limits_{m=t_k-1}^{t_{k-1}}\eta_{m}\sum\limits_{c=1}^N \varrho_c\sum\limits_{j\in\mathcal S_c}\rho_{j,c}\Vert\nabla F_j(\mathbf w_j^{(m)})\Vert
    \nonumber \\&
    +2(1-\alpha_k) \sum\limits_{m=t_k-1}^{t_{k-1}}\eta_{m}\sigma,
\end{align}
where (a) results from Jensen's inequality and (b) comes from Assumption~\ref{assump:SGD_noise}.

Considering the terms on the right hand side of the above equation, we can upper bound $\Vert\nabla F_i(\mathbf w_i^{(t)})\Vert$ as 
\begin{align} \label{eq:expand}
    \Big\Vert\nabla F_i(\mathbf w_i^{(t)})\Big\Vert
    \leq \Big\Vert\nabla F_i(\mathbf w_i^{(t)})-\nabla \hat{F}_c(\bar{\mathbf w}_c^{(t)})\Big\Vert
    + \Big\Vert\nabla \hat{F}_c(\bar{\mathbf w}_c^{(t)})-\nabla F(\mathbf w^*)\Big\Vert.
\end{align}
Moreover, using $\beta$-smoothness, we have 
    \begin{align} \label{eq:beta}
        &\Big\Vert\nabla F_i(\mathbf w_i^{(t)})-\nabla \hat{F}_c(\bar{\mathbf w}_c^{(t)})\Big\Vert
        + \Big\Vert\nabla \hat{F}_c(\bar{\mathbf w}_c^{(t)})-\nabla F(\mathbf w^*)\Big\Vert
        \nonumber \\&
        \leq 
        \beta\Big\Vert\mathbf w_i^{(t)}-\bar{\mathbf w}_c^{(t)}\Big\Vert
        + \beta\Big\Vert\mathbf w_c^{(t)}-\mathbf w^*\Big\Vert.
    \end{align}
Combining~\eqref{eq:expand} and~\eqref{eq:beta} gives us:
\begin{align} \label{eq:combine}
    \Big\Vert\nabla F_i(\mathbf w_i^{(t)})\Big\Vert
    \leq
    \beta\Big\Vert\mathbf w_i^{(t)}-\bar{\mathbf w}_c^{(t)}\Big\Vert
        + \beta\Big\Vert\mathbf w_c^{(t)}-\mathbf w^*\Big\Vert.
\end{align}
Replacing the result of~\eqref{eq:combine} in~\eqref{eq:tri_e} yields:
\begin{align}

    \nonumber \\&
    \leq

    \nonumber \\&
    +\alpha_k\beta\sum\limits_{m=t_k-1}^{t_k-\Delta}\eta_m\sum\limits_{c=1}^N \varrho_c\sum\limits_{j\in\mathcal S_c}\rho_{j,c}\Big(\Big\Vert\mathbf w_j^{(m)}-\bar{\mathbf w}_c^{(m)}\Big\Vert

    \nonumber \\&

        + \Big\Vert\mathbf w_c^{(m)}-\mathbf w^*\Big\Vert\Big)
    \nonumber \\&
    +\alpha_k\sum\limits_{m=t_k-1}^{t_k-\Delta}\eta_m\sigma
    +2(1-\alpha_k) \sum\limits_{m=t_k-1}^{t_{k-1}}\eta_{m}\sigma.
\end{align}
