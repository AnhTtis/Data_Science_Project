\section{Safety and Stability}

\begin{lemma}\label{lemma1}
The probability of survivability ($p_{s_i}$) in non-increasing over subsequent segments of a trajectory. That is, $p_{s_i} \ge p_{s_{i+1}} \; \forall i \in \{0,N-1 \}$ where $N$ is the number of segments of the trajectory.

\begin{proof}
The survivability is given by $p_{s_{i+1}} = (1- p_{c_{i+1}}) \cdot p_{s_{i}}$. Since, $0 \le p_{c_i} \le 1$, $p_{s_{i+1}} \le p_{s_{i}}$.
\end{proof}

\end{lemma}

\begin{lemma}\label{lemma2}
Maximum progress ($d_{p_i}$) or change in distance from obstacles ($d_{o_i}$) over a trajectory segment is bounded by the product of velocity and time step ($|v_i| \cdot h$).
$$|\Delta d_{p_i}| = \alpha_i \cdot |v_i| \cdot h \le |v_i|\cdot h, \quad -1\le\alpha_i\le 1$$
$$|\Delta d_{o_i}| = \beta_i \cdot |v_i| \cdot h \le |v_i|\cdot h, \quad -1\le\beta_i\le 1$$
\end{lemma}


\begin{definition}
Let us define a safe set by,
$$\mathcal{C} = \{ \mathbf{x} \subset D \; \in \; \mathrm{R}^n | h(x) \ge 0 \}$$
$$
\mathcal{C} \subset \mathcal{D} \subset R^n$$
\end{definition}

\begin{definition}
 Let $\mathcal{C} \subset R^n$ be a set defined as the super-level set of a continuously differentiable function $h: \mathcal{D} \subset R^n \rightarrow R$. Then any trajectory $\tau = \min J$ for the system (1) renders set $\mathcal{C}$ invariant. Additionally, set $\mathcal{D}$ is stable. That is, minimizing $J$ in $\mathcal{D}$ either reduces the distance to set $\mathcal{C}$.   

\end{definition}

In addition to the above (common) definition of control barrier functions (CBF). We also define a \em{relaxed} definition for CBF below. 
\begin{definition}
 Let $\mathcal{C} \subset R^n$ be a set defined as the super-level set of a continuously differentiable function $h: \mathcal{D} \subset R^n \rightarrow R$. Then any trajectory $\tau = \min J$ for the system (1) renders set $\mathcal{C}$ invariant. Additionally, set $\mathcal{D}$ is (weakly) stable. That is, minimizing $J$ in $\mathcal{D}$ either retains or reduces the distance to set $\mathcal{C}$.  
 
\end{definition}
